              while IFS= read -r pattern; do

              HAS_SERVICEACCOUNTS=0; HAS_INGRESS=0

              while IFS= read -r line; do

                  pods)           HAS_PODS=1 ;;
                  jobs)           HAS_JOBS=1 ;;
                  cronjobs)       HAS_CRONJOBS=1 ;;

              # -----------------------------------------------------------
              # Helper: safely fetch a workload resource as JSON
              # Returns {"items":[]} on any error
              # -----------------------------------------------------------
              fetch_workload_safe() {
                local full_name="$1"
                local ns="$2"
                local temp_file
                temp_file=$(mktemp)

                if ! oc get "$full_name" -n "$ns" -o json >"$temp_file" 2>/dev/null; then
                  echo '{"items":[]}'
                  rm -f "$temp_file"
                  return
                fi

                if jq -e '(. | type == "object") and (has("items") or has("apiVersion"))' "$temp_file" >/dev/null 2>&1; then
                  cat "$temp_file"
                else
                  echo '{"items":[]}'
                fi

                rm -f "$temp_file"
              }

              # -----------------------------------------------------------
              # Helper: Route TLS secret references (BUG FIX v3.0)
              # Previously missing - was causing Route TLS secrets to be deleted
              # -----------------------------------------------------------
              get_route_references() {
                local ns="$1"
                [[ "$HAS_ROUTES" != "1" ]] && echo "" && return

                oc get routes.route.openshift.io -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    [
                      .items[]? |
                      (.spec.tls?.secretName // empty)
                    ] | unique | .[] | select(length > 0)
                  ' 2>/dev/null || echo ""
              }

              # -----------------------------------------------------------
              # Helper: ServiceAccount pull secret references
              # -----------------------------------------------------------
              get_serviceaccount_references() {
                local ns="$1"
                [[ "$HAS_SERVICEACCOUNTS" != "1" ]] && echo "" && return

                oc get serviceaccounts -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    [
                      .items[]? |
                      (
                        ((.imagePullSecrets // [])[]?.name // empty),
                        ((.secrets // [])[]?.name // empty)
                      )
                    ] | unique | .[] | select(length > 0)
                  ' 2>/dev/null || echo ""
              }

              # -----------------------------------------------------------
              # Helper: Ingress TLS secret references
              # -----------------------------------------------------------
              get_ingress_references() {
                local ns="$1"
                [[ "$HAS_INGRESS" != "1" ]] && echo "" && return

                oc get ingresses.networking.k8s.io -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    [
                      .items[]? |
                      ((.spec.tls // [])[]?.secretName // empty)
                    ] | unique | .[] | select(length > 0)
                  ' 2>/dev/null || echo ""
              }

              # -----------------------------------------------------------
              # Helper: All workload secret references
              # Includes: volumes.secret, volumes.projected.sources[].secret (BUG FIX v3.0),
              #           envFrom.secretRef, env.valueFrom.secretKeyRef
              #           + Routes, ServiceAccounts, Ingress
              # Output format: two sections separated by markers
              #   ---REFS-START--- / ---REFS-END--- : unique secret names (one per line)
              #   remainder: detail lines  secretname:kind:resourcename
              # -----------------------------------------------------------
              get_workload_references() {
                local ns="$1"

                local f_deploy f_sts f_ds f_rs f_job f_cronjob f_pod f_merged
                f_deploy=$(mktemp); f_sts=$(mktemp); f_ds=$(mktemp); f_rs=$(mktemp)
                f_job=$(mktemp);    f_cronjob=$(mktemp); f_pod=$(mktemp); f_merged=$(mktemp)

                cleanup_wl_temps() {
                  rm -f "$f_deploy" "$f_sts" "$f_ds" "$f_rs" "$f_job" "$f_cronjob" "$f_pod" "$f_merged"
                }
                trap cleanup_wl_temps RETURN

                echo '{"items":[]}' > "$f_deploy"
                echo '{"items":[]}' > "$f_sts"
                echo '{"items":[]}' > "$f_ds"
                echo '{"items":[]}' > "$f_rs"
                echo '{"items":[]}' > "$f_job"
                echo '{"items":[]}' > "$f_cronjob"
                echo '{"items":[]}' > "$f_pod"

                [[ "$HAS_DEPLOYMENTS" == "1" ]]  && fetch_workload_safe "deployments.apps"  "$ns" > "$f_deploy"
                [[ "$HAS_STATEFULSETS" == "1" ]]  && fetch_workload_safe "statefulsets.apps"  "$ns" > "$f_sts"
                [[ "$HAS_DAEMONSETS" == "1" ]]    && fetch_workload_safe "daemonsets.apps"    "$ns" > "$f_ds"
                [[ "$HAS_REPLICASETS" == "1" ]]   && fetch_workload_safe "replicasets.apps"   "$ns" > "$f_rs"
                [[ "$HAS_JOBS" == "1" ]]          && fetch_workload_safe "jobs.batch"         "$ns" > "$f_job"
                [[ "$HAS_CRONJOBS" == "1" ]]      && fetch_workload_safe "cronjobs.batch"     "$ns" > "$f_cronjob"
                [[ "$HAS_PODS" == "1" ]]          && fetch_workload_safe "pods"               "$ns" > "$f_pod"

                if ! jq -s '{items: ([.[] | .items // []] | add)}' \
                  "$f_deploy" "$f_sts" "$f_ds" "$f_rs" "$f_job" "$f_cronjob" "$f_pod" \
                  > "$f_merged" 2>/dev/null; then
                  log WARN "Failed to merge workloads in $ns"
                  echo "---REFS-START---"
                  echo "---REFS-END---"
                  return
                fi

                # Unique secret names from workload specs
                # Includes both normal secret volumes AND projected volume sources (BUG FIX v3.0)
                refs=$(jq -r '
                  [
                    .items[]? |
                    (
                      # Deployment / StatefulSet / DaemonSet / ReplicaSet
                      (select(.kind == "Deployment" or .kind == "StatefulSet" or .kind == "DaemonSet" or .kind == "ReplicaSet") |
                        .spec.template.spec |
                        (
                          ((.volumes // [])[]?.secret?.secretName // empty),
                          ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                          ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty),
                          ((.initContainers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.initContainers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                        )
                      ),
                      # CronJob
                      (select(.kind == "CronJob") |
                        .spec.jobTemplate.spec.template.spec |
                        (
                          ((.volumes // [])[]?.secret?.secretName // empty),
                          ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                          ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                        )
                      ),
                      # Job
                      (select(.kind == "Job") |
                        .spec.template.spec |
                        (
                          ((.volumes // [])[]?.secret?.secretName // empty),
                          ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                          ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                        )
                      ),
                      # Pod
                      (select(.kind == "Pod") |
                        .spec |
                        (
                          ((.volumes // [])[]?.secret?.secretName // empty),
                          ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                          ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty),
                          ((.initContainers // [])[]?.envFrom[]?.secretRef?.name // empty),
                          ((.initContainers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                        )
                      )
                    )
                  ] | unique | .[] | select(length > 0)
                ' "$f_merged" 2>/dev/null || echo "")

                # Detail lines for logging: secretname:kind:resourcename
                details=$(jq -r '
                  .items[]? | . as $item |
                  (
                    (select(.kind == "Deployment" or .kind == "StatefulSet" or .kind == "DaemonSet" or .kind == "ReplicaSet") |
                      .spec.template.spec |
                      [
                        ((.volumes // [])[]?.secret?.secretName // empty),
                        ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                        ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty),
                        ((.initContainers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.initContainers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                      ] | .[] | select(length > 0) | "\(.):deployment:\($item.metadata.name)"
                    ),
                    (select(.kind == "CronJob") |
                      .spec.jobTemplate.spec.template.spec |
                      [
                        ((.volumes // [])[]?.secret?.secretName // empty),
                        ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                        ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                      ] | .[] | select(length > 0) | "\(.):cronjob:\($item.metadata.name)"
                    ),
                    (select(.kind == "Job") |
                      .spec.template.spec |
                      [
                        ((.volumes // [])[]?.secret?.secretName // empty),
                        ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                        ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                      ] | .[] | select(length > 0) | "\(.):job:\($item.metadata.name)"
                    ),
                    (select(.kind == "Pod") |
                      .spec |
                      [
                        ((.volumes // [])[]?.secret?.secretName // empty),
                        ((.volumes // [])[]?.projected.sources[]?.secret?.name // empty),
                        ((.containers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.containers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty),
                        ((.initContainers // [])[]?.envFrom[]?.secretRef?.name // empty),
                        ((.initContainers // [])[]?.env[]?.valueFrom?.secretKeyRef?.name // empty)
                      ] | .[] | select(length > 0) | "\(.):pod:\($item.metadata.name)"
                    )
                  )
                ' "$f_merged" 2>/dev/null || echo "")

                route_refs=$(get_route_references "$ns")
                sa_refs=$(get_serviceaccount_references "$ns")
                ingress_refs=$(get_ingress_references "$ns")

                # Merge all unique refs
                all_refs=$(printf '%s\n%s\n%s\n%s\n' "$refs" "$route_refs" "$sa_refs" "$ingress_refs" \
                  | sort -u | grep -v '^$' || echo "")

                # Build additional detail lines for route/sa/ingress refs
                additional_details=""
                while IFS= read -r secret_name; do
                  [[ -z "$secret_name" ]] && continue
                  if echo "$route_refs" | grep -Fxq "$secret_name" 2>/dev/null; then
                    additional_details+="${secret_name}:route:tls"$'\n'
                  fi
                  if echo "$sa_refs" | grep -Fxq "$secret_name" 2>/dev/null; then
                    additional_details+="${secret_name}:serviceaccount:pull-secret"$'\n'
                  fi
                  if echo "$ingress_refs" | grep -Fxq "$secret_name" 2>/dev/null; then
                    additional_details+="${secret_name}:ingress:tls"$'\n'
                  fi
                done <<< "$(printf '%s\n%s\n%s\n' "$route_refs" "$sa_refs" "$ingress_refs" | grep -v '^$' || echo "")"

                combined_details=$(printf '%s\n%s' "$details" "$additional_details" | grep -v '^$' || echo "")

                # Emit structured output
                echo "---REFS-START---"
                echo "$all_refs"
                echo "---REFS-END---"
                echo "$combined_details"
              }

              # -----------------------------------------------------------
              # Encryption-config protection (NEW v3.0)
              # Returns 0 (should protect) or 1 (not protected by this check)
              # -----------------------------------------------------------
              is_encryption_config_protected() {
                local secret_name="$1"
                local all_encryption_secrets="$2"   # newline-separated, sorted newest first

                # Check if it matches the active key in the APIServer CR
                local active_keys
                active_keys=$(oc get apiserver cluster \
                  -o jsonpath='{.spec.encryption.aescbc.keys[*].name}' 2>/dev/null || echo "")

                if [[ -n "$active_keys" ]]; then
                  if echo "$active_keys" | grep -Fq "$secret_name" 2>/dev/null; then
                    log INFO "  PROTECTED: $secret_name (active encryption key in APIServer CR)"
                    return 0
                  fi
                fi

                # Protect the 3 most recently created encryption-config secrets
                local recent_3
                recent_3=$(echo "$all_encryption_secrets" | head -3 || echo "")
                if echo "$recent_3" | grep -Fxq "$secret_name" 2>/dev/null; then
                  log INFO "  PROTECTED: $secret_name (top-3 most recent encryption-config)"
                  return 0
                fi

                return 1
              }

              # -----------------------------------------------------------
              # Core: process_secrets
              # Args:
              #   $1  target namespace
              #   $2  JSON blob of secrets (full oc get secret -o json output)
              #   $3  newline-separated list of referenced secret names
              #   $4  workload detail lines (secretname:kind:resourcename)
              #   $5  effective_dry_run ("true"/"false")
              # Sets globals: ns_deleted, ns_skipped
              # -----------------------------------------------------------
              process_secrets() {
                local target_ns="$1"
                local secret_json="$2"
                local referenced_secrets="$3"
                local workload_details="$4"
                local effective_dry_run="$5"

                local deleted_count=0
                local skipped_count=0
                local batch_counter=0

                # Pre-compute sorted encryption-config list for this namespace
                # (sorted newest first by name, works because names include date/counter suffix)
                local all_encryption_secrets=""
                if [[ "$target_ns" == "openshift-kube-apiserver" ]]; then
                  all_encryption_secrets=$(echo "$secret_json" | \
                    jq -r '.items[]? | select(.metadata.name | startswith("encryption-config-")) | .metadata.name' 2>/dev/null | \
                    sort -r || echo "")

                  local enc_count
                  enc_count=$(echo "$all_encryption_secrets" | grep -c . 2>/dev/null || echo "0")
                  log INFO "  Found $enc_count encryption-config secrets (will protect active + top 3)"
                fi

                while IFS= read -r secret_data; do
                  [[ -z "$secret_data" ]] && continue

                  local name
                  name=$(echo "$secret_data" | jq -r '.name' 2>/dev/null || echo "")
                  [[ -z "$name" || "$name" == "null" ]] && continue

                  local secret_type
                  secret_type=$(echo "$secret_data" | jq -r '.type // "Opaque"' 2>/dev/null || echo "Opaque")

                  # --- Check 1: Excluded by type ---
                  local excluded_by_type=false
                  while IFS= read -r excluded_type; do
                    [[ -z "$excluded_type" ]] && continue
                    if [[ "$secret_type" == "$excluded_type" ]]; then
                      excluded_by_type=true
                      ((skipped_count++)) || true
                      ((total_system_secret++)) || true
                      log INFO "  SKIPPED: $name (type: $secret_type)"
                      break
                    fi
                  done <<< "$EXCLUDED_TYPES"
                  [[ "$excluded_by_type" == "true" ]] && continue

                  # --- Check 2: Protected labels ---
                  local labels
                  labels=$(echo "$secret_data" | jq -r '.labels // {}' 2>/dev/null || echo "{}")

                  local skip=false
                  local matched_label=""
                  while IFS= read -r label_rule; do
                    [[ -z "$label_rule" ]] && continue

                    if [[ "$label_rule" == *"/"* ]] && [[ "$label_rule" != *"="* ]]; then
                      if echo "$labels" | jq -e "to_entries[] | select(.key | startswith(\"${label_rule}\"))" &>/dev/null 2>&1; then
                        skip=true; matched_label="${label_rule}*"
                        ((total_protected++)) || true
                        break
                      fi
                    elif [[ "$label_rule" == *"="* ]]; then
                      local key="${label_rule%=*}"
                      local expected="${label_rule#*=}"
                      local actual
                      actual=$(echo "$labels" | jq -r ".\"$key\" // empty" 2>/dev/null || echo "")
                      if [[ "$actual" == "$expected" ]]; then
                        skip=true; matched_label="$label_rule"
                        ((total_protected++)) || true
                        break
                      fi
                    else
                      if echo "$labels" | jq -e ".\"$label_rule\"" &>/dev/null 2>&1; then
                        skip=true; matched_label="$label_rule"
                        ((total_protected++)) || true
                        break
                      fi
                    fi
                  done <<< "$PROTECTED_LABELS"

                  if [[ "$skip" == "true" ]]; then
                    ((skipped_count++)) || true
                    log INFO "  PROTECTED: $name (label: $matched_label)"
                    continue
                  fi

                  # --- Check 3: Encryption-config protection (openshift-kube-apiserver only) ---
                  if [[ "$target_ns" == "openshift-kube-apiserver" ]] && [[ "$name" =~ ^encryption-config- ]]; then
                    if is_encryption_config_protected "$name" "$all_encryption_secrets"; then
                      ((skipped_count++)) || true
                      ((total_encryption_protected++)) || true
                      continue
                    fi
                  fi

                  # --- Check 4: Age ---
                  local created
                  created=$(echo "$secret_data" | jq -r '.created' 2>/dev/null || echo "")
                  local age_check_passed=true

                  if [[ -n "$created" && "$created" != "null" ]]; then
                    local created_epoch
                    created_epoch=$(date -d "$created" +%s 2>/dev/null || echo "0")
                    if [[ "$created_epoch" -gt 0 ]]; then
                      local age_seconds age_days
                      age_seconds=$(( current_epoch - created_epoch ))
                      age_days=$(( age_seconds / 86400 ))
                      if [[ "$age_seconds" -lt "$min_age_seconds" ]]; then
                        age_check_passed=false
                        ((skipped_count++)) || true
                        ((total_too_new++)) || true
                        log INFO "  TOO-NEW: $name (age: ${age_days}d)"
                      fi
                    else
                      if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                        age_check_passed=false
                        ((skipped_count++)) || true
                        ((total_no_timestamp++)) || true
                        log INFO "  INVALID-TIMESTAMP: $name (skipped)"
                      fi
                    fi
                  else
                    if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                      age_check_passed=false
                      ((skipped_count++)) || true
                      ((total_no_timestamp++)) || true
                      log INFO "  NO-TIMESTAMP: $name (skipped)"
                    fi
                  fi
                  [[ "$age_check_passed" == "false" ]] && continue

                  # --- Check 5: Referenced by workloads/routes/SAs ---
                  if echo "$referenced_secrets" | grep -Fxq "$name" 2>/dev/null; then
                    ((skipped_count++)) || true
                    local workload_ref
                    workload_ref=$(echo "$workload_details" | grep "^${name}:" | head -1 | cut -d: -f2- 2>/dev/null || echo "unknown")

                    if [[ "$workload_ref" == "route:"* ]]; then
                      ((total_route_tls++)) || true
                      log INFO "  REFERENCED: $name (Route TLS)"
                    elif [[ "$workload_ref" == "serviceaccount:"* ]]; then
                      ((total_sa_refs++)) || true
                      log INFO "  REFERENCED: $name (ServiceAccount)"
                    else
                      ((total_referenced++)) || true
                      log INFO "  REFERENCED: $name (by: $workload_ref)"
                    fi
                    continue
                  fi

                  # --- All checks passed: delete or dry-run ---
                  if [[ "$effective_dry_run" == "false" ]]; then
                    # Race-condition guard: re-check references immediately before deletion
                    local current_ref_result
                    current_ref_result=$(get_workload_references "$target_ns")
                    local current_referenced
                    current_referenced=$(echo "$current_ref_result" | \
                      sed -n '/---REFS-START---/,/---REFS-END---/p' | \
                      grep -v "^---REFS-" || echo "")

                    if echo "$current_referenced" | grep -Fxq "$name" 2>/dev/null; then
                      ((skipped_count++)) || true
                      ((total_referenced++)) || true
                      log WARN "  RACE-PREVENTED: $name became referenced, skipping"
                      continue
                    fi

                    sleep "$DELAY_DEL"

                    local deletion_output
                    deletion_output=$(oc delete secret "$name" -n "$target_ns" --wait=false 2>&1)
                    if [[ $? -eq 0 ]]; then
                      ((deleted_count++)) || true
                      ((total_deleted++)) || true
                      ((batch_counter++)) || true
                      deleted_secrets_list+=("${target_ns}/${name}")
                      log INFO "  DELETED: $name"

                      if [[ $(( batch_counter % BATCH_SIZE )) -eq 0 ]]; then
                        sleep "$BATCH_PAUSE"
                      fi
                    else
                      ((total_errors++)) || true
                      local err_msg="Failed to delete $name in $target_ns: $deletion_output"
                      error_list+=("$err_msg")
                      log ERROR "  DELETE-FAILED: $name - $deletion_output"
                    fi
                  else
                    ((deleted_count++)) || true
                    ((total_deleted++)) || true
                    log INFO "  DRY-RUN: Would delete $name"
                  fi

                done < <(echo "$secret_json" | \
                  jq -c '.items[] | {name: .metadata.name, type: .type, labels: .metadata.labels, created: .metadata.creationTimestamp}' \
                  2>/dev/null || echo "")

                ns_deleted=$deleted_count
                ns_skipped=$skipped_count
              }

              # -----------------------------------------------------------
              # High-volume namespace processor (paginated)
              # -----------------------------------------------------------
              process_high_volume_namespace() {
                local ns="$1"
                local total_secret_count="$2"
                local effective_dry_run="$3"

                log INFO "HIGH-VOLUME: $ns ($total_secret_count Secrets) - paginated processing"
                ((namespaces_paginated++)) || true

                local ref_result
                ref_result=$(get_workload_references "$ns")
                local referenced_secrets
                referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                local workload_details
                workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")

                local deleted_in_ns=0
                local skipped_in_ns=0
                local pagination_failed=false
                local continue_token=""
                local page_num=0
                local total_pages
                total_pages=$(( (total_secret_count + PAGINATION_LIMIT - 1) / PAGINATION_LIMIT ))
                local retry_count=0
                local max_retries=3

                log INFO "PAGINATION: ~$total_pages pages ($PAGINATION_LIMIT/page)"

                while true; do
                  ((page_num++)) || true
                  log INFO "PAGE $page_num/$total_pages: Fetching..."

                  local page_result=""
                  retry_count=0
                  local fetch_success=false

                  while [[ $retry_count -lt $max_retries ]]; do
                    page_result=""
                    if [[ -z "$continue_token" ]]; then
                      page_result=$(oc get secret -n "$ns" --limit="$PAGINATION_LIMIT" -o json 2>&1 || echo "")
                    else
                      page_result=$(oc get secret -n "$ns" --limit="$PAGINATION_LIMIT" --continue="$continue_token" -o json 2>&1 || echo "")
                    fi

                    if [[ -n "$page_result" ]] && echo "$page_result" | jq -e '.items' >/dev/null 2>&1; then
                      fetch_success=true
                      break
                    else
                      ((retry_count++)) || true
                      if [[ $retry_count -lt $max_retries ]]; then
                        log WARN "PAGE $page_num: Fetch failed (attempt $retry_count/$max_retries), retrying..."
                        sleep 2
                      fi
                    fi
                  done

                  if [[ "$fetch_success" != "true" ]]; then
                    log WARN "PAGE $page_num: Failed after $max_retries retries"
                    pagination_failed=true
                    break
                  fi

                  if ! echo "$page_result" | jq empty >/dev/null 2>&1; then
                    ((total_json_validation_failures++)) || true

                    if echo "$page_result" | grep -qE "continue token.*expired|too old resource version" 2>/dev/null; then
                      log WARN "PAGE $page_num: Continue token expired"
                      if [[ $retry_count -lt $max_retries ]]; then
                        ((retry_count++)) || true
                        ((total_continue_token_retries++)) || true
                        log WARN "Restarting pagination from beginning (retry $retry_count/$max_retries)"
                        continue_token=""
                        page_num=0
                        deleted_in_ns=0
                        skipped_in_ns=0
                        sleep 2
                        continue
                      else
                        log ERROR "PAGE $page_num: Max retries reached for continue token expiration"
                        error_list+=("Continue token expired and max retries reached in $ns")
                        ((total_errors++)) || true
                        pagination_failed=true
                        break
                      fi
                    else
                      log ERROR "PAGE $page_num: Invalid JSON"
                      error_list+=("Invalid JSON at page $page_num in $ns")
                      ((total_errors++)) || true
                      pagination_failed=true
                      break
                    fi
                  fi

                  log INFO "PAGE $page_num: Processing..."
                  process_secrets "$ns" "$page_result" "$referenced_secrets" "$workload_details" "$effective_dry_run"

                  ((deleted_in_ns += ns_deleted)) || true
                  ((skipped_in_ns += ns_skipped)) || true
                  log INFO "PAGE $page_num: done (deleted=$ns_deleted skipped=$ns_skipped)"

                  continue_token=$(echo "$page_result" | jq -r '.metadata.continue // empty' 2>/dev/null || echo "")
                  if [[ -z "$continue_token" ]]; then
                    log INFO "PAGE $page_num: Last page reached"
                    break
                  fi

                  sleep 0.5
                done

                # Fallback: single full fetch if pagination failed
                if [[ "$pagination_failed" == "true" ]]; then
                  log WARN "PAGINATION FAILED at page $page_num - attempting full fetch fallback"
                  deleted_in_ns=0; skipped_in_ns=0

                  local secret_json=""
                  secret_json=$(oc get secret -n "$ns" -o json 2>&1 || echo "")
                  if [[ -n "$secret_json" ]] && echo "$secret_json" | jq -e '.items' >/dev/null 2>&1; then
                    log INFO "FALLBACK: Full fetch succeeded, processing..."
                    process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    deleted_in_ns=$ns_deleted
                    skipped_in_ns=$ns_skipped
                  else
                    log ERROR "FALLBACK: Failed - both pagination and fallback failed for $ns"
                    error_list+=("Both pagination and fallback failed for $ns")
                    ((total_errors++)) || true
                  fi
                fi

                log INFO "Namespace summary: deleted=$deleted_in_ns skipped=$skipped_in_ns pages=$page_num"
                ((total_skipped += skipped_in_ns)) || true
              }

              # -----------------------------------------------------------
              # Main processing loop
              # -----------------------------------------------------------
              namespace_list=$(oc get ns -o jsonpath='{.items[*].metadata.name}' 2>/dev/null | tr ' ' '\n' | sort || echo "")
              if [[ -z "$namespace_list" ]]; then
                error_exit "No namespaces found or insufficient permissions"
              fi

              namespace_count=$(echo "$namespace_list" | wc -l)
              log INFO "Found $namespace_count namespaces in cluster"

              current_epoch=$(date +%s)
              min_age_seconds=$(( MIN_AGE_DAYS * 86400 ))
              start_time=$(date +%s)

              # Loop-scoped variables initialized here to avoid set -u unbound errors
              included=false
              is_review_only=false
              effective_dry_run="$DRY_RUN"
              elapsed=0
              secret_count=0
              ref_result=""
              referenced_secrets=""
              workload_details=""
              ref_count=0
              secret_json=""
              review_note=""

              while IFS= read -r ns; do
                [[ -z "$ns" ]] && continue

                # --- Step 1: Include check ---
                included=false
                while IFS= read -r pattern; do
                  [[ -z "$pattern" ]] && continue
                  if echo "$ns" | grep -Eq "$pattern" 2>/dev/null; then
                    included=true
                    break
                  fi
                done <<< "$INCLUDED_NS"

                if [[ "$included" == "false" ]]; then
                  ((namespaces_not_included++)) || true
                  continue
                fi

                ((namespaces_processed++)) || true

                # --- Step 2: Review-mode check ---
                is_review_only=false
                while IFS= read -r pattern; do
                  [[ -z "$pattern" ]] && continue
                  if echo "$ns" | grep -Eq "$pattern" 2>/dev/null; then
                    is_review_only=true
                    break
                  fi
                done <<< "$DRYRUN_REVIEW_NS"

                effective_dry_run="$DRY_RUN"
                if [[ "$is_review_only" == "true" ]]; then
                  effective_dry_run="true"
                  ((namespaces_review_mode++)) || true
                fi

                if [[ $((namespaces_processed % 10)) -eq 0 ]]; then
                  elapsed=$(( $(date +%s) - start_time ))
                  log INFO "Progress: $namespaces_processed/${namespace_count} namespaces, ${elapsed}s elapsed"
                fi

                log INFO "-----------------------------------------------"
                log INFO "Processing Namespace [$namespaces_processed]: $ns"
                if [[ "$is_review_only" == "true" ]]; then
                  log INFO "REVIEW-MODE: Namespace in review-only list (forced DRY_RUN)"
                fi
                log INFO "-----------------------------------------------"

                secret_count=0
                if ! secret_count=$(oc get secret -n "$ns" --no-headers 2>/dev/null | wc -l); then
                  log ERROR "Unable to count Secrets in $ns"
                  error_list+=("Unable to count Secrets in: $ns")
                  ((total_errors++)) || true
                  sleep "$DELAY_NS"
                  continue
                fi

                if [[ "$secret_count" -eq 0 ]]; then
                  log INFO "No Secrets found in $ns"
                  sleep "$DELAY_NS"
                  continue
                fi

                log INFO "Found $secret_count Secret(s)"

                # High-volume-only mode: skip normal-volume namespaces
                if [[ "$PROCESS_ONLY_HIGH_VOLUME" == "true" ]]; then
                  if [[ "$secret_count" -le "$HIGH_VOLUME_THRESHOLD" ]]; then
                    log INFO "SKIPPED: high-volume-only mode active (threshold=$HIGH_VOLUME_THRESHOLD)"
                    ((namespaces_skipped_by_mode++)) || true
                    sleep 0.01
                    continue
                  fi
                fi

                # High-volume path (paginated)
                if [[ "$secret_count" -gt "$HIGH_VOLUME_THRESHOLD" ]]; then
                  if [[ "$PROCESS_HIGH_VOLUME" == "true" ]]; then
                    process_high_volume_namespace "$ns" "$secret_count" "$effective_dry_run"
                  else
                    log WARN "HIGH-VOLUME: $ns skipped (set processHighVolumeNamespaces=true to enable)"
                    high_volume_ns_list+=("${ns}:${secret_count}")
                    ((namespaces_high_volume++)) || true
                  fi
                  sleep "$DELAY_NS"
                  continue
                fi

                # Normal path
                ref_result=$(get_workload_references "$ns")
                referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")

                ref_count=$(echo "$referenced_secrets" | grep -c . 2>/dev/null || echo "0")
                log INFO "Extracted $ref_count unique secret references from workloads/routes/SAs"

                secret_json=""
                if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                  log ERROR "Unable to list Secrets in $ns"
                  error_list+=("Unable to list Secrets in: $ns")
                  ((total_errors++)) || true
                  sleep "$DELAY_NS"
                  continue
                fi

                process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"

                review_note=""
                [[ "$is_review_only" == "true" ]] && review_note=" (review mode - no actual deletions)"
                log INFO "Namespace summary: deleted=$ns_deleted skipped=$ns_skipped${review_note}"
                ((total_skipped += ns_skipped)) || true

                sleep "$DELAY_NS"

              done <<< "$namespace_list"

              # -----------------------------------------------------------
              # Final summary
              # -----------------------------------------------------------
              end_time=$(date +%s)
              duration=$(( end_time - start_time ))
              duration_min=$(( duration / 60 ))
              duration_sec=$(( duration % 60 ))
              run_mode=$(if [[ "$DRY_RUN" == "true" ]]; then echo "DRY-RUN"; else echo "LIVE"; fi)

              log INFO ""
              log INFO "================================================"
              log INFO "Job Completed"
              log INFO "================================================"
              log INFO "Duration: ${duration_min}m ${duration_sec}s"
              log INFO ""
              log INFO "Namespace Statistics:"
              log INFO "  Total in cluster:          $namespace_count"
              log INFO "  Not in include list:       $namespaces_not_included"
              log INFO "  Processed:                 $namespaces_processed"
              log INFO "    - Review-mode (dry-run): $namespaces_review_mode"
              log INFO "    - High-volume paginated: $namespaces_paginated"
              log INFO "    - High-volume skipped:   $namespaces_high_volume"
              if [[ "$PROCESS_ONLY_HIGH_VOLUME" == "true" ]]; then
                log INFO "    - Skipped (hv-only mode): $namespaces_skipped_by_mode"
              fi
              log INFO ""

              if [[ "${#high_volume_ns_list[@]}" -gt 0 ]]; then
                log WARN "High-Volume Namespaces Skipped (set processHighVolumeNamespaces=true):"
                for entry in "${high_volume_ns_list[@]}"; do
                  log WARN "  - ${entry%:*}: ${entry#*:} Secrets"
                done
                log WARN ""
              fi

              log INFO "Secret Statistics:"
              log INFO "  Deleted (or would-delete):     $total_deleted"
              log INFO "  Skipped total:                 $total_skipped"
              log INFO "    - System types:              $total_system_secret"
              log INFO "    - Protected labels:          $total_protected"
              log INFO "    - Too new (<${MIN_AGE_DAYS}d): $total_too_new"
              log INFO "    - No/invalid timestamp:      $total_no_timestamp"
              log INFO "    - Referenced (workloads):    $total_referenced"
              log INFO "    - Referenced (Route TLS):    $total_route_tls"
              log INFO "    - Referenced (ServiceAccts): $total_sa_refs"
              log INFO "    - Encryption-config protect: $total_encryption_protected"
              log INFO ""

              if [[ "$PROCESS_HIGH_VOLUME" == "true" ]]; then
                log INFO "Pagination Statistics:"
                log INFO "  Continue-token retries:      $total_continue_token_retries"
                log INFO "  JSON validation failures:    $total_json_validation_failures"
                log INFO ""
              fi

              if [[ "$DRY_RUN" == "false" && "${#deleted_secrets_list[@]}" -gt 0 ]]; then
                log INFO "=== DELETED SECRETS (${#deleted_secrets_list[@]} total) ==="
                for deleted_secret in "${deleted_secrets_list[@]}"; do
                  log INFO "  $deleted_secret"
                done
                log INFO ""
              fi

              if [[ "$total_errors" -gt 0 ]]; then
                log ERROR "Errors: $total_errors"
                for err_msg in "${error_list[@]}"; do
                  log ERROR "  $err_msg"
                done
                log ERROR ""
              else
                log INFO "Errors: 0"
              fi

              log INFO "Global mode: $run_mode"
              log INFO "Namespace mode: INCLUDE-BASED"
              log INFO "Review-only namespaces forced to DRY-RUN: $namespaces_review_mode"
              log INFO "================================================"

              exit 0

                    [[ "$keep_count" == "0" ]] && return 1

                        select(.metadata.name | test($pattern)) |
                        (.metadata.creationTimestamp | fromdateiso8601) as $epoch |
                        "\($epoch)|\(.metadata.name)"
                      ' | sort -nr | head -n "$keep_count" | cut -d'|' -f2)
                    
                    if echo "$all_matching" | grep -Fxq "$secret_name" >/dev/null 2>&1; then
                      return 0
                    fi
                  fi

              process_secrets() {
                local target_ns="$1"
                local secret_json="$2"
                local referenced_secrets="$3"
                local workload_details="$4"
                local effective_dry_run="${5:-$DRY_RUN}"
                local deleted_count=0
                local skipped_count=0
                local batch_counter=0
                
                while IFS= read -r secret_data; do
                  [[ -z "$secret_data" ]] && continue
                  
                  name=$(echo "$secret_data" | jq -r '.name' 2>/dev/null)
                  [[ -z "$name" || "$name" == "null" ]] && continue
                  
                  secret_type=$(echo "$secret_data" | jq -r '.type // "Opaque"' 2>/dev/null)
                  
                  excluded_by_type=false
                  while IFS= read -r excluded_type; do
                    [[ -z "$excluded_type" ]] && continue
                    if [[ "$secret_type" == "$excluded_type" ]]; then
                      excluded_by_type=true
                      ((skipped_count++))
                      ((total_system_secret++))
                      log INFO "SKIPPED: $name (type: $secret_type)"
                      break
                    fi
                  done <<< "$EXCLUDED_TYPES"
                  [[ "$excluded_by_type" == "true" ]] && continue
                  
                  excluded_by_name=false
                  while IFS= read -r excluded_pattern; do
                    [[ -z "$excluded_pattern" ]] && continue
                    if echo "$name" | grep -Eq "$excluded_pattern" 2>/dev/null; then
                      excluded_by_name=true
                      ((skipped_count++))
                      ((total_excluded_names++))
                      log INFO "PROTECTED: $name (name pattern: $excluded_pattern)"
                      break
                    fi
                  done <<< "$EXCLUDED_SECRET_NAMES"
                  [[ "$excluded_by_name" == "true" ]] && continue
                  
                  if should_retain_recent_rotation "$name" "$target_ns"; then
                    ((skipped_count++))
                    ((total_protected++))
                    log INFO "RETAINED: $name (recent rotation - retention policy)"
                    continue
                  fi
                  
                  labels=$(echo "$secret_data" | jq -r '.labels // {}' 2>/dev/null)
                  
                  skip=false
                  matched_label=""
                  while IFS= read -r label_rule; do
                    [[ -z "$label_rule" ]] && continue
                    
                    if [[ "$label_rule" == *"/"* ]] && [[ "$label_rule" != *"="* ]]; then
                      if echo "$labels" | jq -e "to_entries[] | select(.key | startswith(\"${label_rule}\"))" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule*"
                        ((total_protected++))
                        break
                      fi
                    elif [[ "$label_rule" == *"="* ]]; then
                      key="${label_rule%=*}"
                      expected="${label_rule#*=}"
                      actual=$(echo "$labels" | jq -r ".\"$key\" // empty" 2>/dev/null)
                      if [[ "$actual" == "$expected" ]]; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    else
                      if echo "$labels" | jq -e ".\"$label_rule\"" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    fi
                  done <<< "$PROTECTED_LABELS"
                  
                  if [[ "$skip" == "true" ]]; then
                    ((skipped_count++))
                    log INFO "PROTECTED: $name (label: $matched_label)"
                    continue
                  fi
                  
                  created=$(echo "$secret_data" | jq -r '.created' 2>/dev/null)
                  age_check_passed=true
                  
                  if [[ -n "$created" && "$created" != "null" ]]; then
                    created_epoch=$(date -d "$created" +%s 2>/dev/null || echo "0")
                    if [[ "$created_epoch" -gt 0 ]]; then
                      age_seconds=$((current_epoch - created_epoch))
                      age_days=$((age_seconds / 86400))
                      
                      if [[ "$age_seconds" -lt "$min_age_seconds" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_too_new++))
                        log INFO "TOO-NEW: $name (age: ${age_days} days)"
                      fi
                    else
                      if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_no_timestamp++))
                        log INFO "INVALID-TIMESTAMP: $name (skipped due to config)"
                      fi
                    fi
                  else
                    if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                      age_check_passed=false
                      ((skipped_count++))
                      ((total_no_timestamp++))
                      log INFO "NO-TIMESTAMP: $name (skipped due to config)"
                    fi
                  fi
                  
                  if [[ "$age_check_passed" == "false" ]]; then
                    continue
                  fi
                  
                  if echo "$referenced_secrets" | grep -Fxq "$name" >/dev/null 2>&1; then
                    ((skipped_count++))
                    
                    workload_ref=$(echo "$workload_details" | grep "^${name}:" | head -1 | cut -d: -f2- || echo "unknown")
                    
                    if [[ "$workload_ref" == "route:"* ]]; then
                      ((total_route_tls++))
                      log INFO "REFERENCED: $name (route TLS)"
                    elif [[ "$workload_ref" == "serviceaccount:"* ]]; then
                      ((total_sa_refs++))
                      log INFO "REFERENCED: $name (service account)"
                    else
                      ((total_referenced++))
                      log INFO "REFERENCED: $name (workload)"
                    fi
                    continue
                  fi
                  
                  if [[ "$effective_dry_run" == "true" ]]; then
                    log INFO "DRY-RUN: Would delete $name (unreferenced, age OK)"
                    deleted_secrets_list+=("[$target_ns] $name")
                  else
                    if oc get secret "$name" -n "$target_ns" &>/dev/null; then
                      if oc delete secret "$name" -n "$target_ns" 2>&1; then
                        log INFO "DELETED: $name"
                        deleted_secrets_list+=("[$target_ns] $name")
                      else
                        log ERROR "FAILED TO DELETE: $name"
                        error_list+=("Failed to delete secret: $name in $target_ns")
                        ((total_errors++))
                        continue
                      fi
                    else
                      log INFO "ALREADY-GONE: $name (deleted during processing)"
                      continue
                    fi
                  fi
                  
                  ((deleted_count++))
                  ((batch_counter++))
                  
                  if [[ $((batch_counter % BATCH_SIZE)) -eq 0 ]]; then
                    sleep "$BATCH_PAUSE"
                  fi
                  
                done < <(echo "$secret_json" | jq -c --arg ns "$target_ns" '
                  .items[] |
                  {
                    name: .metadata.name,
                    type: .type,
                    created: .metadata.creationTimestamp,
                    labels: .metadata.labels
                  }
                ')
                
                ns_deleted=$deleted_count
                ns_skipped=$skipped_count
              }
              
              DELAY_NS=0.5
              BATCH_SIZE="${BATCH_SIZE:-10}"
              BATCH_PAUSE="${BATCH_PAUSE:-1}"
              
              all_namespaces=$(oc get namespaces -o jsonpath='{.items[*].metadata.name}')
              
              while IFS= read -r ns; do
                [[ -z "$ns" ]] && continue
                
                CURRENT_NS="$ns"
                
                included=false
                while IFS= read -r pattern; do
                  [[ -z "$pattern" ]] && continue
                  if echo "$ns" | grep -Eq "$pattern" 2>/dev/null; then
                    included=true
                    break
                  fi
                done <<< "$INCLUDED_NS"
                
                if [[ "$included" == "false" ]]; then
                  ((namespaces_not_included++))
                  continue
                fi
                
                ((namespaces_processed++))
                
                # Check if this namespace is in review-only mode (forced DRY_RUN)
                force_dry_run=false
                if [[ -n "$DRY_RUN_REVIEW_NAMESPACES" ]]; then
                  while IFS= read -r review_pattern; do
                    [[ -z "$review_pattern" ]] && continue
                    if [[ "$ns" =~ $review_pattern ]]; then
                      force_dry_run=true
                      break
                    fi
                  done <<< "$DRY_RUN_REVIEW_NAMESPACES"
                fi
                
                # Determine effective dry-run mode (forced or global)
                effective_dry_run="$DRY_RUN"
                if [[ "$force_dry_run" == "true" ]]; then
                  effective_dry_run="true"
                  log INFO "REVIEW-MODE: Namespace in review-only list (forced DRY_RUN)"
                fi
                
                if [[ $((namespaces_processed % 50)) -eq 0 ]]; then
                  elapsed=$(($(date +%s) - start_time))
                  log INFO "Progress: Processed $namespaces_processed namespaces in ${elapsed}s"
                fi
                
                log INFO "Processing Namespace [$namespaces_processed]: $ns"
                
                total_secret_count=$(oc get secrets -n "$ns" --no-headers 2>/dev/null | wc -l || echo "0")
                log INFO "Found $total_secret_count Secret(s)"
                
                if [[ "$PROCESS_HIGH_VOLUME" == "true" ]] && [[ "$total_secret_count" -ge "$HIGH_VOLUME_THRESHOLD" ]]; then
                  log INFO "HIGH-VOLUME: Enabling pagination ($PAGINATION_LIMIT per page)"
                  ((namespaces_high_volume++))
                  
                  ((namespaces_paginated++))
                  
                  ref_result=$(get_workload_references "$ns")
                  referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                  workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")
                  
                  # Debug logging for reference extraction
                  ref_count=$(echo "$referenced_secrets" | grep -v '^$' | wc -l)
                  log INFO "Extracted $ref_count unique secret references from workloads"
                  
                  crd_refs=$(get_crd_references "$ns")
                  if [[ -n "$crd_refs" ]]; then
                    referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                  fi
                  
                  local deleted_in_ns=0
                  local skipped_in_ns=0
                  local pagination_failed=false
                  local continue_token=""
                  local page_num=0
                  local total_pages=$(( (total_secret_count + PAGINATION_LIMIT - 1) / PAGINATION_LIMIT ))
                  local retry_count=0
                  local max_retries=3
                  
                  log INFO "PAGINATION: Processing ~$total_pages pages ($PAGINATION_LIMIT per page)"
                  
                  while true; do
                    ((page_num++))
                    log INFO "PAGE $page_num/$total_pages: Fetching..."
                    
                    local fetch_cmd="oc get secrets -n \"$ns\" --limit=\"$PAGINATION_LIMIT\""
                    [[ -n "$continue_token" ]] && fetch_cmd+=" --continue=\"$continue_token\""
                    fetch_cmd+=" -o json 2>&1"
                    
                    local fetch_result=$(eval "$fetch_cmd")
                    local fetch_exit=$?
                    
                    if [[ $fetch_exit -ne 0 ]]; then
                      if echo "$fetch_result" | grep -qi "continue token.*expired" >/dev/null 2>&1; then
                        log WARN "PAGE $page_num: Continue token expired"
                        ((total_continue_token_retries++))
                        
                        if [[ $retry_count -lt $max_retries ]]; then
                          ((retry_count++))
                          log INFO "RETRY $retry_count/$max_retries: Restarting pagination from beginning"
                          continue_token=""
                          page_num=0
                          sleep 2
                          continue
                        else
                          log ERROR "PAGE $page_num: Max retries reached, falling back to non-paginated"
                          pagination_failed=true
                          break
                        fi
                      else
                        log ERROR "PAGE $page_num: Failed to fetch secrets: $fetch_result"
                        pagination_failed=true
                        break
                      fi
                    fi
                    
                    if ! echo "$fetch_result" | jq -e '.items' &>/dev/null; then
                      log ERROR "PAGE $page_num: Invalid JSON response"
                      ((total_json_validation_failures++))
                      pagination_failed=true
                      break
                    fi
                    
                    local page_secret_count=$(echo "$fetch_result" | jq '.items | length' 2>/dev/null || echo "0")
                    log INFO "PAGE $page_num: Processing $page_secret_count secrets..."
                    
                    process_secrets "$ns" "$fetch_result" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    ((deleted_in_ns += ns_deleted))
                    ((skipped_in_ns += ns_skipped))
                    
                    local next_token=$(echo "$fetch_result" | jq -r '.metadata.continue // empty' 2>/dev/null)
                    
                    if [[ -z "$next_token" ]]; then
                      log INFO "PAGINATION: Complete ($page_num pages processed)"
                      break
                    fi
                    
                    continue_token="$next_token"
                    retry_count=0
                  done
                  
                  if [[ "$pagination_failed" == "true" ]]; then
                    log WARN "FALLBACK: Processing namespace without pagination"
                    deleted_in_ns=0
                    skipped_in_ns=0
                    
                    secret_json=""
                    if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                      log ERROR "Unable to list Secrets in $ns"
                      error_list+=("Unable to list Secrets in namespace: $ns")
                      ((total_errors++))
                      sleep "$DELAY_NS"
                      continue
                    fi
                    
                    process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    deleted_in_ns=$ns_deleted
                    skipped_in_ns=$ns_skipped
                  fi
                  
                  log INFO "Namespace summary: deleted=$deleted_in_ns skipped=$skipped_in_ns"
                  ((total_deleted += deleted_in_ns))
                  ((total_skipped += skipped_in_ns))
                  
                  sleep "$DELAY_NS"
                  continue
                fi
                
                ref_result=$(get_workload_references "$ns")
                referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")
                
                # Debug logging for reference extraction
                ref_count=$(echo "$referenced_secrets" | grep -v '^$' | wc -l)
                log INFO "Extracted $ref_count unique secret references from workloads"
                
                crd_refs=$(get_crd_references "$ns")
                if [[ -n "$crd_refs" ]]; then
                  referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                fi
                
                secret_json=""
                if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                  log ERROR "Unable to list Secrets in $ns"
                  error_list+=("Unable to list Secrets in namespace: $ns")
                  ((total_errors++))
                  sleep "$DELAY_NS"
                  continue
                fi
                
                process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                
                log INFO "Namespace summary: deleted=$ns_deleted skipped=$ns_skipped"
                ((total_skipped += ns_skipped))
                ((total_deleted += ns_deleted))
                
                sleep "$DELAY_NS"
              done <<< "$all_namespaces"
              
              CURRENT_NS=""
              
              end_time=$(date +%s)
              duration=$((end_time - start_time))
              
              log INFO "=================================="
              log INFO "Secret Pruner Summary"
              log INFO "=================================="
              log INFO "Namespace Statistics:"
              log INFO "  Processed: $namespaces_processed"
              log INFO "  Not included: $namespaces_not_included"
              log INFO "  High-volume: $namespaces_high_volume"
              log INFO "  Paginated: $namespaces_paginated"
              log INFO ""
              log INFO "Secret Statistics:"
              log INFO "  Deleted (or would delete): $total_deleted"
              log INFO "  Skipped total: $total_skipped"
              log INFO "    - System (types): $total_system_secret"
              log INFO "    - System (names): $total_excluded_names"
              log INFO "    - Protected (labels): $total_protected"
              log INFO "    - Too new (< $MIN_AGE_DAYS days): $total_too_new"
              log INFO "    - Referenced (workloads): $total_referenced"
              log INFO "    - Referenced (routes): $total_route_tls"
              log INFO "    - Referenced (service accounts): $total_sa_refs"
              log INFO "    - No timestamp: $total_no_timestamp"
              log INFO ""
              log INFO "Performance:"
              log INFO "  Duration: ${duration}s"
              log INFO "  Continue token retries: $total_continue_token_retries"
              log INFO "  JSON validation failures: $total_json_validation_failures"
              log INFO "  Errors: $total_errors"
              log INFO ""
              
              if [[ ${#deleted_secrets_list[@]} -gt 0 ]]; then
                log INFO "Deleted Secrets ($total_deleted):"
                for deleted in "${deleted_secrets_list[@]}"; do
                  log INFO "  - $deleted"
                done
              fi
              
              if [[ ${#error_list[@]} -gt 0 ]]; then
                log ERROR "Errors encountered ($total_errors):"
                for error in "${error_list[@]}"; do
                  log ERROR "  - $error"
                done
              fi
              
              log INFO "=================================="
              log INFO "Completed successfully"
              exit 0

                  return
                fi
                
                oc get routes.route.openshift.io -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    .items[]? |
                    (.spec.tls?.secretName // empty)
                  ' 2>/dev/null | { grep -v '^$' || true; }
              }
              
              get_sa_references() {
                local ns="$1"
                
                oc get serviceaccounts -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    [
                      .items[]? |
                      ((.secrets // [])[]? | .name // empty),
                      ((.imagePullSecrets // [])[]? | .name // empty)
                    ] | unique | .[] | select(length > 0)
                  ' 2>/dev/null || echo ""
              }
              
              get_ingress_references() {
                local ns="$1"
                
                if [[ "$HAS_INGRESS" != "1" ]]; then
                  echo ""
                  return
                fi
                
                oc get ingresses.networking.k8s.io -n "$ns" -o json 2>/dev/null | \
                  jq -r '
                    [
                      .items[]? |
                      ((.spec.tls // [])[]?.secretName // empty)
                    ] | unique | .[] | select(length > 0)
                  ' 2>/dev/null || echo ""
              }
              
              get_crd_references() {
                local ns="$1"
                local all_refs=""
                
                # Only 3 CRDs needed for openshift-kube-apiserver
                local critical_crds=(
                  "apiservers.config.openshift.io"
                  "oauths.config.openshift.io"
                  "authentications.config.openshift.io"
                )
                
                local jq_extract='
                  .items[]? |
                  (.spec.credentialsSecretName // empty),
                  (.spec.signingSecrets[]?.name // empty),
                  (.spec.encryption.aescbc.keys[]?.name // empty),
                  (.spec.tlsSecurityProfile.custom.certificate.name // empty),
                  (.spec.trustedCA.name // empty),
                  (.spec.clients[]?.secret // empty),
                  (.spec.defaultCertificate.name // empty),
                  (.. | objects | to_entries[] |
                    select(.key |
                      endswith("Secret") or endswith("secret") or
                      endswith("SecretName") or endswith("secretName") or
                      endswith("SecretRef") or endswith("secretRef") or
                      test("^(secret|password|credential|ca|tls|cert|kubeconfig|kubeConfig)"; "i")
                    ) |
                    if .value | type == "object" then .value.name // empty
                    elif .value | type == "string" then .value
                    else empty end
                  )
                '
                
                for crd in "${critical_crds[@]}"; do
                  # Check if CRD exists
                  crd_exists=$(oc api-resources 2>/dev/null | grep -q "${crd%.*}" && echo "1" || echo "0")
                  if [[ "$crd_exists" != "1" ]]; then
                    continue
                  fi
                  
                  if [[ -n "$ns" ]]; then
                    # Check if CRD is namespaced
                    is_namespaced=$(oc api-resources --namespaced=true 2>/dev/null | grep -q "${crd%.*}" && echo "1" || echo "0")
                    if [[ "$is_namespaced" == "1" ]]; then
                    local ns_refs=$(oc get "$crd" -n "$ns" -o json 2>/dev/null | \
                      jq -r "$jq_extract" 2>/dev/null | grep -v '^$')
                    
                    if [[ -n "$ns_refs" ]]; then
                      all_refs+="$ns_refs"$'\n'
                    fi
                    fi
                  fi
                  
                  # Check if CRD is cluster-scoped
                  is_cluster=$(oc api-resources --namespaced=false 2>/dev/null | grep -q "${crd%.*}" && echo "1" || echo "0")
                  if [[ "$is_cluster" == "1" ]]; then
                    local cluster_refs=$(oc get "$crd" -o json 2>/dev/null | \
                      jq -r "$jq_extract" 2>/dev/null | grep -v '^$')
                    
                    if [[ -n "$cluster_refs" ]]; then
                      all_refs+="$cluster_refs"$'\n'
                    fi
                  fi
                done
                
                echo "$all_refs" | grep -v '^$' | sort -u
              }
              
              should_retain_recent_rotation() {
                local secret_name="$1"
                local ns="$2"
                
                [[ -z "$RETAIN_RECENT_ROTATIONS" ]] && return 1
                
                while IFS=':' read -r pattern keep_count; do
                  [[ -z "$pattern" || -z "$keep_count" ]] && continue
                  keep_count=$(echo "$keep_count" | tr -d ' ')
                  
                  if [[ "$secret_name" =~ ^${pattern} ]]; then
                    [[ "$keep_count" == "0" ]] && return 1
                    
                    # Sort by epoch timestamp (most recent first)
                    local all_matching=$(oc get secrets -n "$ns" -o json 2>/dev/null | \
                      jq -r --arg pattern "^${pattern}" '
                        .items[] |
                        select(.metadata.name | test($pattern)) |
                        (.metadata.creationTimestamp | fromdateiso8601) as $epoch |
                        "\($epoch)|\(.metadata.name)"
                      ' | sort -nr | head -n "$keep_count" | cut -d'|' -f2)
                    
                    if echo "$all_matching" | grep -Fxq "$secret_name" >/dev/null 2>&1; then
                      return 0
                    fi
                  fi
                done <<< "$RETAIN_RECENT_ROTATIONS"
                
                return 1
              }
              
              get_workload_references() {
                local ns="$1"
                
                local f_deploy=$(mktemp)
                local f_sts=$(mktemp)
                local f_ds=$(mktemp)
                local f_rs=$(mktemp)
                local f_job=$(mktemp)
                local f_cronjob=$(mktemp)
                local f_pod=$(mktemp)
                local f_merged=$(mktemp)
                
                cleanup_temp_files() {
                  rm -f "$f_deploy" "$f_sts" "$f_ds" "$f_rs" "$f_job" "$f_cronjob" "$f_pod" "$f_merged"
                }
                trap cleanup_temp_files RETURN
                
                echo '{"items":[]}' > "$f_deploy"
                echo '{"items":[]}' > "$f_sts"
                echo '{"items":[]}' > "$f_ds"
                echo '{"items":[]}' > "$f_rs"
                echo '{"items":[]}' > "$f_job"
                echo '{"items":[]}' > "$f_cronjob"
                echo '{"items":[]}' > "$f_pod"
                
                [[ "$HAS_DEPLOYMENTS" == "1" ]] && fetch_workload_safe "deployments.apps" "$ns" > "$f_deploy"
                [[ "$HAS_STATEFULSETS" == "1" ]] && fetch_workload_safe "statefulsets.apps" "$ns" > "$f_sts"
                [[ "$HAS_DAEMONSETS" == "1" ]] && fetch_workload_safe "daemonsets.apps" "$ns" > "$f_ds"
                [[ "$HAS_REPLICASETS" == "1" ]] && fetch_workload_safe "replicasets.apps" "$ns" > "$f_rs"
                [[ "$HAS_JOBS" == "1" ]] && fetch_workload_safe "jobs.batch" "$ns" > "$f_job"
                [[ "$HAS_CRONJOBS" == "1" ]] && fetch_workload_safe "cronjobs.batch" "$ns" > "$f_cronjob"
                fetch_workload_safe "pods" "$ns" > "$f_pod"
                
                jq -s '
                  {items: ([.[].items[]?] | unique_by(.metadata.uid // .metadata.name))}
                ' "$f_deploy" "$f_sts" "$f_ds" "$f_rs" "$f_job" "$f_cronjob" "$f_pod" > "$f_merged"
                
                local secret_refs=$(jq -r '
                  .items[]? |
                  .metadata.name as $workload_name |
                  .kind // "Unknown" as $workload_kind |
                  [
                    (.spec.template.spec.volumes // [])[]? | select(.secret != null) | .secret.secretName,
                    (.spec.volumes // [])[]? | select(.secret != null) | .secret.secretName,
                    (.spec.template.spec.volumes // [])[]? | select(.projected != null) | (.projected.sources // [])[]? | select(.secret != null) | .secret.name,
                    (.spec.volumes // [])[]? | select(.projected != null) | (.projected.sources // [])[]? | select(.secret != null) | .secret.name,
                    (.spec.template.spec.containers // [])[]? | (.env // [])[]? | select(.valueFrom.secretKeyRef != null) | .valueFrom.secretKeyRef.name,
                    (.spec.containers // [])[]? | (.env // [])[]? | select(.valueFrom.secretKeyRef != null) | .valueFrom.secretKeyRef.name,
                    (.spec.template.spec.containers // [])[]? | (.envFrom // [])[]? | select(.secretRef != null) | .secretRef.name,
                    (.spec.containers // [])[]? | (.envFrom // [])[]? | select(.secretRef != null) | .secretRef.name,
                    (.spec.template.spec.imagePullSecrets // [])[]?.name,
                    (.spec.imagePullSecrets // [])[]?.name
                  ] |
                  map(select(. != null and . != "")) |
                  unique[] |
                  "\(.):\($workload_kind):\($workload_name)"
                ' "$f_merged" 2>/dev/null || echo "")
                
                local route_refs=$(get_route_references "$ns")
                local sa_refs=$(get_sa_references "$ns")
                local ingress_refs=$(get_ingress_references "$ns")
                
                local all_workload_refs=$(echo "$secret_refs" | cut -d: -f1 | sort -u)
                local all_route_refs=$(echo "$route_refs" | sort -u)
                local all_sa_refs=$(echo "$sa_refs" | sort -u)
                local all_ingress_refs=$(echo "$ingress_refs" | sort -u)
                
                local all_unique=$(printf '%s\n%s\n%s\n%s\n' "$all_workload_refs" "$all_route_refs" "$all_sa_refs" "$all_ingress_refs" | \
                  grep -v '^$' | sort -u)
                
                echo "---REFS-START---"
                echo "$all_unique"
                echo "---REFS-END---"
                echo "$secret_refs" | while IFS=: read -r secret_name kind workload_name; do
                  [[ -z "$secret_name" ]] && continue
                  echo "$secret_name:workload:$kind/$workload_name"
                done
                
                echo "$route_refs" | while read -r secret_name; do
                  [[ -z "$secret_name" ]] && continue
                  echo "$secret_name:route:tls-certificate"
                done
                
                echo "$sa_refs" | while read -r secret_name; do
                  [[ -z "$secret_name" ]] && continue
                  echo "$secret_name:serviceaccount:image-pull-or-token"
                done
                
                echo "$ingress_refs" | while read -r secret_name; do
                  [[ -z "$secret_name" ]] && continue
                  echo "$secret_name:ingress:tls-certificate"
                done
              }
              
              process_secrets() {
                local target_ns="$1"
                local secret_json="$2"
                local referenced_secrets="$3"
                local workload_details="$4"
                local effective_dry_run="${5:-$DRY_RUN}"
                local deleted_count=0
                local skipped_count=0
                local batch_counter=0
                
                while IFS= read -r secret_data; do
                  [[ -z "$secret_data" ]] && continue
                  
                  name=$(echo "$secret_data" | jq -r '.name' 2>/dev/null)
                  [[ -z "$name" || "$name" == "null" ]] && continue
                  
                  secret_type=$(echo "$secret_data" | jq -r '.type // "Opaque"' 2>/dev/null)
                  
                  excluded_by_type=false
                  while IFS= read -r excluded_type; do
                    [[ -z "$excluded_type" ]] && continue
                    if [[ "$secret_type" == "$excluded_type" ]]; then
                      excluded_by_type=true
                      ((skipped_count++))
                      ((total_system_secret++))
                      log INFO "SKIPPED: $name (type: $secret_type)"
                      break
                    fi
                  done <<< "$EXCLUDED_TYPES"
                  [[ "$excluded_by_type" == "true" ]] && continue
                  
                  excluded_by_name=false
                  while IFS= read -r excluded_pattern; do
                    [[ -z "$excluded_pattern" ]] && continue
                    if [[ "$name" =~ $excluded_pattern ]]; then
                      excluded_by_name=true
                      ((skipped_count++))
                      ((total_excluded_names++))
                      log INFO "PROTECTED: $name (name pattern: $excluded_pattern)"
                      break
                    fi
                  done <<< "$EXCLUDED_SECRET_NAMES"
                  [[ "$excluded_by_name" == "true" ]] && continue
                  
                  if should_retain_recent_rotation "$name" "$target_ns"; then
                    ((skipped_count++))
                    ((total_protected++))
                    log INFO "RETAINED: $name (recent rotation - retention policy)"
                    continue
                  fi
                  
                  labels=$(echo "$secret_data" | jq -r '.labels // {}' 2>/dev/null)
                  
                  skip=false
                  matched_label=""
                  while IFS= read -r label_rule; do
                    [[ -z "$label_rule" ]] && continue
                    
                    if [[ "$label_rule" == *"/"* ]] && [[ "$label_rule" != *"="* ]]; then
                      if echo "$labels" | jq -e "to_entries[] | select(.key | startswith(\"${label_rule}\"))" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule*"
                        ((total_protected++))
                        break
                      fi
                    elif [[ "$label_rule" == *"="* ]]; then
                      key="${label_rule%=*}"
                      expected="${label_rule#*=}"
                      actual=$(echo "$labels" | jq -r ".\"$key\" // empty" 2>/dev/null)
                      if [[ "$actual" == "$expected" ]]; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    else
                      if echo "$labels" | jq -e ".\"$label_rule\"" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    fi
                  done <<< "$PROTECTED_LABELS"
                  
                  if [[ "$skip" == "true" ]]; then
                    ((skipped_count++))
                    log INFO "PROTECTED: $name (label: $matched_label)"
                    continue
                  fi
                  
                  created=$(echo "$secret_data" | jq -r '.created' 2>/dev/null)
                  age_check_passed=true
                  
                  if [[ -n "$created" && "$created" != "null" ]]; then
                    created_epoch=$(date -d "$created" +%s 2>/dev/null || echo "0")
                    if [[ "$created_epoch" -gt 0 ]]; then
                      age_seconds=$((current_epoch - created_epoch))
                      age_days=$((age_seconds / 86400))
                      
                      if [[ "$age_seconds" -lt "$min_age_seconds" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_too_new++))
                        log INFO "TOO-NEW: $name (age: ${age_days} days)"
                      fi
                    else
                      if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_no_timestamp++))
                        log INFO "INVALID-TIMESTAMP: $name (skipped due to config)"
                      fi
                    fi
                  else
                    if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                      age_check_passed=false
                      ((skipped_count++))
                      ((total_no_timestamp++))
                      log INFO "NO-TIMESTAMP: $name (skipped due to config)"
                    fi
                  fi
                  
                  if [[ "$age_check_passed" == "false" ]]; then
                    continue
                  fi
                  
                  if echo "$referenced_secrets" | grep -Fxq "$name" >/dev/null 2>&1; then
                    ((skipped_count++))
                    
                    workload_ref=$(echo "$workload_details" | grep "^${name}:" | head -1 | cut -d: -f2- || echo "unknown")
                    
                    if [[ "$workload_ref" == "route:"* ]]; then
                      ((total_route_tls++))
                      log INFO "REFERENCED: $name (route TLS)"
                    elif [[ "$workload_ref" == "serviceaccount:"* ]]; then
                      ((total_sa_refs++))
                      log INFO "REFERENCED: $name (service account)"
                    else
                      ((total_referenced++))
                      log INFO "REFERENCED: $name (workload)"
                    fi
                    continue
                  fi
                  
                  if [[ "$effective_dry_run" == "true" ]]; then
                    log INFO "DRY-RUN: Would delete $name (unreferenced, age OK)"
                    deleted_secrets_list+=("[$target_ns] $name")
                  else
                    if oc get secret "$name" -n "$target_ns" &>/dev/null; then
                      if oc delete secret "$name" -n "$target_ns" 2>&1; then
                        log INFO "DELETED: $name"
                        deleted_secrets_list+=("[$target_ns] $name")
                      else
                        log ERROR "FAILED TO DELETE: $name"
                        error_list+=("Failed to delete secret: $name in $target_ns")
                        ((total_errors++))
                        continue
                      fi
                    else
                      log INFO "ALREADY-GONE: $name (deleted during processing)"
                      continue
                    fi
                  fi
                  
                  ((deleted_count++))
                  ((batch_counter++))
                  
                  if [[ $((batch_counter % BATCH_SIZE)) -eq 0 ]]; then
                    sleep "$BATCH_PAUSE"
                  fi
                  
                done < <(echo "$secret_json" | jq -c --arg ns "$target_ns" '
                  .items[] |
                  {
                    name: .metadata.name,
                    type: .type,
                    created: .metadata.creationTimestamp,
                    labels: .metadata.labels
                  }
                ')
                
                ns_deleted=$deleted_count
                ns_skipped=$skipped_count
              }
              
              DELAY_NS=0.5
              BATCH_SIZE="${BATCH_SIZE:-10}"
              BATCH_PAUSE="${BATCH_PAUSE:-1}"
              
              all_namespaces=$(oc get namespaces -o jsonpath='{.items[*].metadata.name}')
              
              while IFS= read -r ns; do
                [[ -z "$ns" ]] && continue
                
                CURRENT_NS="$ns"
                
                included=false
                while IFS= read -r pattern; do
                  [[ -z "$pattern" ]] && continue
                  if [[ "$ns" =~ $pattern ]]; then
                    included=true
                    break
                  fi
                done <<< "$INCLUDED_NS"
                
                if [[ "$included" == "false" ]]; then
                  ((namespaces_not_included++))
                  continue
                fi
                
                ((namespaces_processed++))
                
                # Check if this namespace is in review-only mode (forced DRY_RUN)
                force_dry_run=false
                if [[ -n "$DRY_RUN_REVIEW_NAMESPACES" ]]; then
                  while IFS= read -r review_pattern; do
                    [[ -z "$review_pattern" ]] && continue
                    if [[ "$ns" =~ $review_pattern ]]; then
                      force_dry_run=true
                      break
                    fi
                  done <<< "$DRY_RUN_REVIEW_NAMESPACES"
                fi
                
                # Determine effective dry-run mode (forced or global)
                effective_dry_run="$DRY_RUN"
                if [[ "$force_dry_run" == "true" ]]; then
                  effective_dry_run="true"
                  log INFO "REVIEW-MODE: Namespace in review-only list (forced DRY_RUN)"
                fi
                
                if [[ $((namespaces_processed % 50)) -eq 0 ]]; then
                  elapsed=$(($(date +%s) - start_time))
                  log INFO "Progress: Processed $namespaces_processed namespaces in ${elapsed}s"
                fi
                
                log INFO "Processing Namespace [$namespaces_processed]: $ns"
                
                total_secret_count=$(oc get secrets -n "$ns" --no-headers 2>/dev/null | wc -l || echo "0")
                log INFO "Found $total_secret_count Secret(s)"
                
                if [[ "$PROCESS_HIGH_VOLUME" == "true" ]] && [[ "$total_secret_count" -ge "$HIGH_VOLUME_THRESHOLD" ]]; then
                  log INFO "HIGH-VOLUME: Enabling pagination ($PAGINATION_LIMIT per page)"
                  ((namespaces_high_volume++))
                  
                  ((namespaces_paginated++))
                  
                  ref_result=$(get_workload_references "$ns")
                  referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | { grep -v "^---REFS-" || true; })
                  workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | { tail -n +2 || true; })
                  
                  # Debug logging for reference extraction
                  ref_count=$(echo "$referenced_secrets" | grep -v '^$' | wc -l)
                  log INFO "Extracted $ref_count unique secret references from workloads"
                  
                  crd_refs=$(get_crd_references "$ns")
                  if [[ -n "$crd_refs" ]]; then
                    referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                  fi
                  
                  local deleted_in_ns=0
                  local skipped_in_ns=0
                  local pagination_failed=false
                  local continue_token=""
                  local page_num=0
                  local total_pages=$(( (total_secret_count + PAGINATION_LIMIT - 1) / PAGINATION_LIMIT ))
                  local retry_count=0
                  local max_retries=3
                  
                  log INFO "PAGINATION: Processing ~$total_pages pages ($PAGINATION_LIMIT per page)"
                  
                  while true; do
                    ((page_num++))
                    log INFO "PAGE $page_num/$total_pages: Fetching..."
                    
                    local fetch_cmd="oc get secrets -n \"$ns\" --limit=\"$PAGINATION_LIMIT\""
                    [[ -n "$continue_token" ]] && fetch_cmd+=" --continue=\"$continue_token\""
                    fetch_cmd+=" -o json 2>&1"
                    
                    local fetch_result=$(eval "$fetch_cmd")
                    local fetch_exit=$?
                    
                    if [[ $fetch_exit -ne 0 ]]; then
                      if echo "$fetch_result" | grep -qi "continue token.*expired" >/dev/null 2>&1; then
                        log WARN "PAGE $page_num: Continue token expired"
                        ((total_continue_token_retries++))
                        
                        if [[ $retry_count -lt $max_retries ]]; then
                          ((retry_count++))
                          log INFO "RETRY $retry_count/$max_retries: Restarting pagination from beginning"
                          continue_token=""
                          page_num=0
                          sleep 2
                          continue
                        else
                          log ERROR "PAGE $page_num: Max retries reached, falling back to non-paginated"
                          pagination_failed=true
                          break
                        fi
                      else
                        log ERROR "PAGE $page_num: Failed to fetch secrets: $fetch_result"
                        pagination_failed=true
                        break
                      fi
                    fi
                    
                    if ! echo "$fetch_result" | jq -e '.items' &>/dev/null; then
                      log ERROR "PAGE $page_num: Invalid JSON response"
                      ((total_json_validation_failures++))
                      pagination_failed=true
                      break
                    fi
                    
                    local page_secret_count=$(echo "$fetch_result" | jq '.items | length' 2>/dev/null || echo "0")
                    log INFO "PAGE $page_num: Processing $page_secret_count secrets..."
                    
                    process_secrets "$ns" "$fetch_result" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    ((deleted_in_ns += ns_deleted))
                    ((skipped_in_ns += ns_skipped))
                    
                    local next_token=$(echo "$fetch_result" | jq -r '.metadata.continue // empty' 2>/dev/null)
                    
                    if [[ -z "$next_token" ]]; then
                      log INFO "PAGINATION: Complete ($page_num pages processed)"
                      break
                    fi
                    
                    continue_token="$next_token"
                    retry_count=0
                  done
                  
                  if [[ "$pagination_failed" == "true" ]]; then
                    log WARN "FALLBACK: Processing namespace without pagination"
                    deleted_in_ns=0
                    skipped_in_ns=0
                    
                    secret_json=""
                    if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                      log ERROR "Unable to list Secrets in $ns"
                      error_list+=("Unable to list Secrets in namespace: $ns")
                      ((total_errors++))
                      sleep "$DELAY_NS"
                      continue
                    fi
                    
                    process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    deleted_in_ns=$ns_deleted
                    skipped_in_ns=$ns_skipped
                  fi
                  
                  log INFO "Namespace summary: deleted=$deleted_in_ns skipped=$skipped_in_ns"
                  ((total_deleted += deleted_in_ns))
                  ((total_skipped += skipped_in_ns))
                  
                  sleep "$DELAY_NS"
                  continue
                fi
                
                ref_result=$(get_workload_references "$ns")
                referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | { grep -v "^---REFS-" || true; })
                workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | { tail -n +2 || true; })
                
                # Debug logging for reference extraction
                ref_count=$(echo "$referenced_secrets" | grep -v '^$' | wc -l)
                log INFO "Extracted $ref_count unique secret references from workloads"
                
                crd_refs=$(get_crd_references "$ns")
                if [[ -n "$crd_refs" ]]; then
                  referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                fi
                
                secret_json=""
                if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                  log ERROR "Unable to list Secrets in $ns"
                  error_list+=("Unable to list Secrets in namespace: $ns")
                  ((total_errors++))
                  sleep "$DELAY_NS"
                  continue
                fi
                
                process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                
                log INFO "Namespace summary: deleted=$ns_deleted skipped=$ns_skipped"
                ((total_skipped += ns_skipped))
                ((total_deleted += ns_deleted))
                
                sleep "$DELAY_NS"
              done <<< "$all_namespaces"
              
              CURRENT_NS=""
              
              end_time=$(date +%s)
              duration=$((end_time - start_time))
              
              log INFO "=================================="
              log INFO "Secret Pruner Summary"
              log INFO "=================================="
              log INFO "Namespace Statistics:"
              log INFO "  Processed: $namespaces_processed"
              log INFO "  Not included: $namespaces_not_included"
              log INFO "  High-volume: $namespaces_high_volume"
              log INFO "  Paginated: $namespaces_paginated"
              log INFO ""
              log INFO "Secret Statistics:"
              log INFO "  Deleted (or would delete): $total_deleted"
              log INFO "  Skipped total: $total_skipped"
              log INFO "    - System (types): $total_system_secret"
              log INFO "    - System (names): $total_excluded_names"
              log INFO "    - Protected (labels): $total_protected"

                  name=$(echo "$secret_data" | jq -r '.name' 2>/dev/null)

                      excluded_by_type=true
                      ((skipped_count++))
                      ((total_system_secret++))
                      log INFO "SKIPPED: $name (type: $secret_type)"
                      break
                    fi
                  done <<< "$EXCLUDED_TYPES"
                  [[ "$excluded_by_type" == "true" ]] && continue
                  
                  excluded_by_name=false
                  while IFS= read -r excluded_pattern; do
                    [[ -z "$excluded_pattern" ]] && continue
                    if [[ "$name" =~ $excluded_pattern ]]; then
                      excluded_by_name=true
                      ((skipped_count++))
                      ((total_excluded_names++))
                      log INFO "PROTECTED: $name (name pattern: $excluded_pattern)"
                      break
                    fi
                  done <<< "$EXCLUDED_SECRET_NAMES"
                  [[ "$excluded_by_name" == "true" ]] && continue
                  
                  if should_retain_recent_rotation "$name" "$target_ns"; then
                    ((skipped_count++))
                    ((total_protected++))
                    log INFO "RETAINED: $name (recent rotation - retention policy)"
                    continue
                  fi
                  
                  labels=$(echo "$secret_data" | jq -r '.labels // {}' 2>/dev/null)
                  
                  skip=false
                  matched_label=""
                  while IFS= read -r label_rule; do
                    [[ -z "$label_rule" ]] && continue
                    
                    if [[ "$label_rule" == *"/"* ]] && [[ "$label_rule" != *"="* ]]; then
                      if echo "$labels" | jq -e "to_entries[] | select(.key | startswith(\"${label_rule}\"))" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule*"
                        ((total_protected++))
                        break
                      fi
                    elif [[ "$label_rule" == *"="* ]]; then
                      key="${label_rule%=*}"
                      expected="${label_rule#*=}"
                      actual=$(echo "$labels" | jq -r ".\"$key\" // empty" 2>/dev/null)
                      if [[ "$actual" == "$expected" ]]; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    else
                      if echo "$labels" | jq -e ".\"$label_rule\"" &>/dev/null; then
                        skip=true
                        matched_label="$label_rule"
                        ((total_protected++))
                        break
                      fi
                    fi
                  done <<< "$PROTECTED_LABELS"
                  
                  if [[ "$skip" == "true" ]]; then
                    ((skipped_count++))
                    log INFO "PROTECTED: $name (label: $matched_label)"
                    continue
                  fi
                  
                  created=$(echo "$secret_data" | jq -r '.created' 2>/dev/null)
                  age_check_passed=true
                  
                  if [[ -n "$created" && "$created" != "null" ]]; then
                    created_epoch=$(date -d "$created" +%s 2>/dev/null || echo "0")
                    if [[ "$created_epoch" -gt 0 ]]; then
                      age_seconds=$((current_epoch - created_epoch))
                      age_days=$((age_seconds / 86400))
                      
                      if [[ "$age_seconds" -lt "$min_age_seconds" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_too_new++))
                        log INFO "TOO-NEW: $name (age: ${age_days} days)"
                      fi
                    else
                      if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                        age_check_passed=false
                        ((skipped_count++))
                        ((total_no_timestamp++))
                        log INFO "INVALID-TIMESTAMP: $name (skipped due to config)"
                      fi
                    fi
                  else
                    if [[ "$PRUNE_WITHOUT_TIMESTAMP" == "false" ]]; then
                      age_check_passed=false
                      ((skipped_count++))
                      ((total_no_timestamp++))
                      log INFO "NO-TIMESTAMP: $name (skipped due to config)"
                    fi
                  fi
                  
                  if [[ "$age_check_passed" == "false" ]]; then
                    continue
                  fi
                  
                  if echo "$referenced_secrets" | grep -Fxq "$name" >/dev/null 2>&1; then
                    ((skipped_count++))
                    
                    workload_ref=$(echo "$workload_details" | grep "^${name}:" | head -1 | cut -d: -f2- || echo "unknown")
                    
                    if [[ "$workload_ref" == "route:"* ]]; then
                      ((total_route_tls++))
                      log INFO "REFERENCED: $name (route TLS)"
                    elif [[ "$workload_ref" == "serviceaccount:"* ]]; then
                      ((total_sa_refs++))
                      log INFO "REFERENCED: $name (service account)"
                    else
                      ((total_referenced++))
                      log INFO "REFERENCED: $name (workload)"
                    fi
                    continue
                  fi
                  
                  if [[ "$effective_dry_run" == "true" ]]; then
                    log INFO "DRY-RUN: Would delete $name (unreferenced, age OK)"
                    deleted_secrets_list+=("[$target_ns] $name")
                  else
                    if oc get secret "$name" -n "$target_ns" &>/dev/null; then
                      if oc delete secret "$name" -n "$target_ns" 2>&1; then
                        log INFO "DELETED: $name"
                        deleted_secrets_list+=("[$target_ns] $name")
                      else
                        log ERROR "FAILED TO DELETE: $name"
                        error_list+=("Failed to delete secret: $name in $target_ns")
                        ((total_errors++))
                        continue
                      fi
                    else
                      log INFO "ALREADY-GONE: $name (deleted during processing)"
                      continue
                    fi
                  fi
                  
                  ((deleted_count++))
                  ((batch_counter++))
                  
                  if [[ $((batch_counter % BATCH_SIZE)) -eq 0 ]]; then
                    sleep "$BATCH_PAUSE"
                  fi
                  
                done < <(echo "$secret_json" | jq -c --arg ns "$target_ns" '
                  .items[] |
                  {
                    name: .metadata.name,
                    type: .type,
                    created: .metadata.creationTimestamp,
                    labels: .metadata.labels
                  }
                ')
                
                ns_deleted=$deleted_count
                ns_skipped=$skipped_count
              }
              
              DELAY_NS=0.5
              BATCH_SIZE="${BATCH_SIZE:-10}"
              BATCH_PAUSE="${BATCH_PAUSE:-1}"
              
              all_namespaces=$(oc get namespaces -o jsonpath='{.items[*].metadata.name}')
              
              while IFS= read -r ns; do
                [[ -z "$ns" ]] && continue
                
                CURRENT_NS="$ns"
                
                included=false
                while IFS= read -r pattern; do
                  [[ -z "$pattern" ]] && continue
                  if [[ "$ns" =~ $pattern ]]; then
                    included=true
                    break
                  fi
                done <<< "$INCLUDED_NS"
                
                if [[ "$included" == "false" ]]; then
                  ((namespaces_not_included++))
                  continue
                fi
                
                ((namespaces_processed++))
                
                # Check if this namespace is in review-only mode (forced DRY_RUN)
                force_dry_run=false
                if [[ -n "$DRY_RUN_REVIEW_NAMESPACES" ]]; then
                  while IFS= read -r review_pattern; do
                    [[ -z "$review_pattern" ]] && continue
                    if [[ "$ns" =~ $review_pattern ]]; then
                      force_dry_run=true
                      break
                    fi
                  done <<< "$DRY_RUN_REVIEW_NAMESPACES"
                fi
                
                # Determine effective dry-run mode (forced or global)
                effective_dry_run="$DRY_RUN"
                if [[ "$force_dry_run" == "true" ]]; then
                  effective_dry_run="true"
                  log INFO "REVIEW-MODE: Namespace in review-only list (forced DRY_RUN)"
                fi
                
                if [[ $((namespaces_processed % 50)) -eq 0 ]]; then
                  elapsed=$(($(date +%s) - start_time))
                  log INFO "Progress: Processed $namespaces_processed namespaces in ${elapsed}s"
                fi
                
                log INFO "Processing Namespace [$namespaces_processed]: $ns"
                
                total_secret_count=$(oc get secrets -n "$ns" --no-headers 2>/dev/null | wc -l || echo "0")
                log INFO "Found $total_secret_count Secret(s)"
                
                if [[ "$PROCESS_HIGH_VOLUME" == "true" ]] && [[ "$total_secret_count" -ge "$HIGH_VOLUME_THRESHOLD" ]]; then
                  log INFO "HIGH-VOLUME: Enabling pagination ($PAGINATION_LIMIT per page)"
                  ((namespaces_high_volume++))
                  
                  ((namespaces_paginated++))
                  
                  ref_result=$(get_workload_references "$ns")
                  referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                  workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")
                  
                  crd_refs=$(get_crd_references "$ns")
                  if [[ -n "$crd_refs" ]]; then
                    referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                  fi
                  
                  local deleted_in_ns=0
                  local skipped_in_ns=0
                  local pagination_failed=false
                  local continue_token=""
                  local page_num=0
                  local total_pages=$(( (total_secret_count + PAGINATION_LIMIT - 1) / PAGINATION_LIMIT ))
                  local retry_count=0
                  local max_retries=3
                  
                  log INFO "PAGINATION: Processing ~$total_pages pages ($PAGINATION_LIMIT per page)"
                  
                  while true; do
                    ((page_num++))
                    log INFO "PAGE $page_num/$total_pages: Fetching..."
                    
                    local fetch_cmd="oc get secrets -n \"$ns\" --limit=\"$PAGINATION_LIMIT\""
                    [[ -n "$continue_token" ]] && fetch_cmd+=" --continue=\"$continue_token\""
                    fetch_cmd+=" -o json 2>&1"
                    
                    local fetch_result=$(eval "$fetch_cmd")
                    local fetch_exit=$?
                    
                    if [[ $fetch_exit -ne 0 ]]; then
                      if echo "$fetch_result" | grep -qi "continue token.*expired" >/dev/null 2>&1; then
                        log WARN "PAGE $page_num: Continue token expired"
                        ((total_continue_token_retries++))
                        
                        if [[ $retry_count -lt $max_retries ]]; then
                          ((retry_count++))
                          log INFO "RETRY $retry_count/$max_retries: Restarting pagination from beginning"
                          continue_token=""
                          page_num=0
                          sleep 2
                          continue
                        else
                          log ERROR "PAGE $page_num: Max retries reached, falling back to non-paginated"
                          pagination_failed=true
                          break
                        fi
                      else
                        log ERROR "PAGE $page_num: Failed to fetch secrets: $fetch_result"
                        pagination_failed=true
                        break
                      fi
                    fi
                    
                    if ! echo "$fetch_result" | jq -e '.items' &>/dev/null; then
                      log ERROR "PAGE $page_num: Invalid JSON response"
                      ((total_json_validation_failures++))
                      pagination_failed=true
                      break
                    fi
                    
                    local page_secret_count=$(echo "$fetch_result" | jq '.items | length' 2>/dev/null || echo "0")
                    log INFO "PAGE $page_num: Processing $page_secret_count secrets..."
                    
                    process_secrets "$ns" "$fetch_result" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    ((deleted_in_ns += ns_deleted))
                    ((skipped_in_ns += ns_skipped))
                    
                    local next_token=$(echo "$fetch_result" | jq -r '.metadata.continue // empty' 2>/dev/null)
                    
                    if [[ -z "$next_token" ]]; then
                      log INFO "PAGINATION: Complete ($page_num pages processed)"
                      break
                    fi
                    
                    continue_token="$next_token"
                    retry_count=0
                  done
                  
                  if [[ "$pagination_failed" == "true" ]]; then
                    log WARN "FALLBACK: Processing namespace without pagination"
                    deleted_in_ns=0
                    skipped_in_ns=0
                    
                    secret_json=""
                    if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                      log ERROR "Unable to list Secrets in $ns"
                      error_list+=("Unable to list Secrets in namespace: $ns")
                      ((total_errors++))
                      sleep "$DELAY_NS"
                      continue
                    fi
                    
                    process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                    
                    deleted_in_ns=$ns_deleted
                    skipped_in_ns=$ns_skipped
                  fi
                  
                  log INFO "Namespace summary: deleted=$deleted_in_ns skipped=$skipped_in_ns"
                  ((total_deleted += deleted_in_ns))
                  ((total_skipped += skipped_in_ns))
                  
                  sleep "$DELAY_NS"
                  continue
                fi
                
                ref_result=$(get_workload_references "$ns")
                referenced_secrets=$(echo "$ref_result" | sed -n '/---REFS-START---/,/---REFS-END---/p' | grep -v "^---REFS-" || echo "")
                workload_details=$(echo "$ref_result" | sed -n '/---REFS-END---/,$p' | tail -n +2 || echo "")
                
                crd_refs=$(get_crd_references "$ns")
                if [[ -n "$crd_refs" ]]; then
                  referenced_secrets=$(printf '%s\n%s' "$referenced_secrets" "$crd_refs" | sort -u)
                fi
                
                secret_json=""
                if ! secret_json=$(oc get secret -n "$ns" -o json 2>/dev/null); then
                  log ERROR "Unable to list Secrets in $ns"
                  error_list+=("Unable to list Secrets in namespace: $ns")
                  ((total_errors++))
                  sleep "$DELAY_NS"
                  continue
                fi
                
                process_secrets "$ns" "$secret_json" "$referenced_secrets" "$workload_details" "$effective_dry_run"
                
                log INFO "Namespace summary: deleted=$ns_deleted skipped=$ns_skipped"
                ((total_skipped += ns_skipped))
                ((total_deleted += ns_deleted))
                
                sleep "$DELAY_NS"
              done <<< "$all_namespaces"
              
              CURRENT_NS=""
              
              end_time=$(date +%s)
              duration=$((end_time - start_time))
              
              log INFO "=================================="
              log INFO "Secret Pruner Summary"
              log INFO "=================================="
              log INFO "Namespace Statistics:"
              log INFO "  Processed: $namespaces_processed"
              log INFO "  Not included: $namespaces_not_included"
              log INFO "  High-volume: $namespaces_high_volume"
              log INFO "  Paginated: $namespaces_paginated"
              log INFO ""
              log INFO "Secret Statistics:"
              log INFO "  Deleted (or would delete): $total_deleted"
              log INFO "  Skipped total: $total_skipped"
              log INFO "    - System (types): $total_system_secret"
              log INFO "    - System (names): $total_excluded_names"
              log INFO "    - Protected (labels): $total_protected"
              log INFO "    - Too new (< $MIN_AGE_DAYS days): $total_too_new"
              log INFO "    - Referenced (workloads): $total_referenced"
              log INFO "    - Referenced (routes): $total_route_tls"
              log INFO "    - Referenced (service accounts): $total_sa_refs"
              log INFO "    - No timestamp: $total_no_timestamp"
              log INFO ""
              log INFO "Performance:"
              log INFO "  Duration: ${duration}s"
              log INFO "  Continue token retries: $total_continue_token_retries"
              log INFO "  JSON validation failures: $total_json_validation_failures"
              log INFO "  Errors: $total_errors"
              log INFO ""
              
              if [[ ${#deleted_secrets_list[@]} -gt 0 ]]; then
                log INFO "Deleted Secrets ($total_deleted):"
                for deleted in "${deleted_secrets_list[@]}"; do
                  log INFO "  - $deleted"
                done
              fi
              
              if [[ ${#error_list[@]} -gt 0 ]]; then
                log ERROR "Errors encountered ($total_errors):"
                for error in "${error_list[@]}"; do
                  log ERROR "  - $error"
                done
              fi
              
              log INFO "=================================="
              log INFO "Completed successfully"
              exit 0
