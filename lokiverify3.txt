#         or re-mirror from quay.io if the current one is already corrupted)
HTTPS_PROXY=http://PROXY_IP:3128 \

  docker://quay.io/rhdh/plugin-catalog-index:1.9 \
  dir:/tmp/catalog-rebuild/catalog-raw

# Step 2: Extract contents

for layer in /tmp/catalog-rebuild/catalog-raw/*; do
  if [ -f "${layer}" ] && ! echo "${layer}" | grep -qE "(manifest\.json|version)$"; then
    tar -xf "${layer}" -C /tmp/catalog-rebuild/catalog-content 2>/dev/null || true
    if [ -f "/tmp/catalog-rebuild/catalog-content/dynamic-plugins.default.yaml" ]; then
      echo "Found catalog content in layer: $(basename ${layer})"
      break
    fi
  fi
done

# Step 3: Rewrite ONLY ghcr.io paths to correct Artifactory path
# ghcr.io/redhat-developer/... -> artifactory.company-xxx.com/image/rhdh/redhat-developer/...
# This preserves the full path including redhat-developer/rhdh-plugin-export-overlays/

sed -i 's|ghcr.io/|artifactory.company-xxx.com/image/rhdh/|g' \
  /tmp/catalog-rebuild/catalog-content/dynamic-plugins.default.yaml

# Verify the replacement looks right
grep "artifactory" /tmp/catalog-rebuild/catalog-content/dynamic-plugins.default.yaml | head -5

# Should show:
# oci://artifactory.company-xxx.com/image/rhdh/redhat-developer/rhdh-plugin-export-overlays/backstage-community-plugin-...

# Also update index.json if present
if [ -f "/tmp/catalog-rebuild/catalog-content/index.json" ]; then
  sed -i 's|ghcr.io/|artifactory.company-xxx.com/image/rhdh/|g' \
    /tmp/catalog-rebuild/catalog-content/index.json
fi

# Also update catalog-entities YAMLs if present
find /tmp/catalog-rebuild/catalog-content/catalog-entities -name "*.yaml" -type f 2>/dev/null | \
  xargs -r sed -i 's|ghcr.io/|artifactory.company-xxx.com/image/rhdh/|g'

# Step 4: Rebuild the catalog index image
mkdir -p /tmp/catalog-rebuild/build/content
cp -r /tmp/catalog-rebuild/catalog-content/* /tmp/catalog-rebuild/build/content/

cat > /tmp/catalog-rebuild/build/Dockerfile << 'EOF'
FROM scratch
COPY content/ /
EOF

podman build -t localhost/rhdh-catalog-index-rebuild:1.9 /tmp/catalog-rebuild/build

# Step 5: Push rebuilt catalog index to Artifactory
skopeo copy --all --dest-tls-verify=false \
  containers-storage:localhost/rhdh-catalog-index-rebuild:1.9 \
  docker://artifactory.company-xxx.com/image/rhdh/rhdh/plugin-catalog-index:1.9

# Cleanup
podman rmi localhost/rhdh-catalog-index-rebuild:1.9 2>/dev/null
rm -rf /tmp/catalog-rebuild

#!/bin/bash
# =============================================================================
# incorporate-plugins-into-catalog-index.sh
#
# Purpose:
#   1. Push manually downloaded ghcr.io plugin OCI images to Artifactory
#   2. Pull the existing catalog index from Artifactory
#   3. Rewrite the ghcr.io references in dynamic-plugins.default.yaml
#      and index.json to point to Artifactory
#   4. Rebuild and push the updated catalog index back to Artifactory
#
# Result: Single catalog index in Artifactory that covers BOTH Red Hat
#         plugins (already there) AND the ghcr.io plugins you added.
#         The init container only talks to Artifactory.
#
# Prerequisites:
#   - skopeo, jq, podman, sed
#   - Authenticated to Artifactory (skopeo login)
#   - 4 ArgoCD plugin images already downloaded to PLUGIN_DIR
#     (using skopeo copy on any internet-connected machine)
#
# How to download the 4 plugins (on any machine with ghcr.io access):
#
#   mkdir -p /tmp/ghcr-plugins/argocd-backend
#   mkdir -p /tmp/ghcr-plugins/argocd-frontend
#   mkdir -p /tmp/ghcr-plugins/roadie-argocd-backend
#   mkdir -p /tmp/ghcr-plugins/roadie-scaffolder-argocd
#
#   skopeo copy --all --src-no-creds \
#     docker://ghcr.io/redhat-developer/rhdh-plugin-export-overlays/backstage-community-plugin-argocd-backend:bs_1.45.3__1.0.2 \
#     dir:/tmp/ghcr-plugins/argocd-backend
#
#   skopeo copy --all --src-no-creds \
#     docker://ghcr.io/redhat-developer/rhdh-plugin-export-overlays/backstage-community-plugin-argocd:bs_1.45.3__2.4.3 \
#     dir:/tmp/ghcr-plugins/argocd-frontend
#
#   skopeo copy --all --src-no-creds \
#     docker://ghcr.io/redhat-developer/rhdh-plugin-export-overlays/roadiehq-backstage-plugin-argo-cd-backend:bs_1.45.3__4.6.0 \
#     dir:/tmp/ghcr-plugins/roadie-argocd-backend
#
#   skopeo copy --all --src-no-creds \
#     docker://ghcr.io/redhat-developer/rhdh-plugin-export-overlays/roadiehq-scaffolder-backend-argocd:bs_1.45.3__1.8.1 \
#     dir:/tmp/ghcr-plugins/roadie-scaffolder-argocd
#
#   Then scp /tmp/ghcr-plugins to bastion and run this script.
# =============================================================================
set -euo pipefail

# ---- VARIABLES -- EDIT THESE ------------------------------------------------
ARTIFACTORY="artifactory.company-xxx.com"
ARTIFACTORY_REPO="rhdh"

# Where the catalog index lives in Artifactory (already mirrored)
CATALOG_INDEX="${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh/plugin-catalog-index:1.9"

# Where the downloaded ghcr.io plugin dirs are
PLUGIN_DIR="/tmp/ghcr-plugins"

# Temp working directory
WORKDIR="/tmp/catalog-index-rebuild"
# ---- END VARIABLES ----------------------------------------------------------

# Plugin mapping: local_dir -> artifactory_path:tag
# Format: DIR_NAME|ARTIFACTORY_IMAGE_PATH
PLUGINS=(
  "argocd-backend|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/backstage-community-plugin-argocd-backend:bs_1.45.3__1.0.2"
  "argocd-frontend|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/backstage-community-plugin-argocd:bs_1.45.3__2.4.3"
  "roadie-argocd-backend|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/roadiehq-backstage-plugin-argo-cd-backend:bs_1.45.3__4.6.0"
  "roadie-scaffolder-argocd|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/roadiehq-scaffolder-backend-argocd:bs_1.45.3__1.8.1"
)

rm -rf "${WORKDIR}"
mkdir -p "${WORKDIR}"

# =============================================================================
# STEP 1: Push plugin images from local dirs to Artifactory
# =============================================================================
printf "=== Step 1: Pushing plugin images to Artifactory ===\n\n"

PUSH_FAILED=0
for entry in "${PLUGINS[@]}"; do
  dir_name="${entry%%|*}"
  target_image="${entry##*|}"
  src_dir="${PLUGIN_DIR}/${dir_name}"

  if [ ! -d "${src_dir}" ]; then
    printf "  [ERROR] Directory not found: %s\n" "${src_dir}"
    printf "          Download it first (see script header for commands)\n\n"
    PUSH_FAILED=1
    continue
  fi

  if [ ! -f "${src_dir}/manifest.json" ]; then
    printf "  [ERROR] No manifest.json in %s -- incomplete download\n\n" "${src_dir}"
    PUSH_FAILED=1
    continue
  fi

  printf "  Pushing: %s\n" "${dir_name}"
  printf "      To:  %s\n" "${target_image}"

  if skopeo copy --all --dest-tls-verify=false \
    "dir:${src_dir}" "docker://${target_image}" 2>&1; then
    printf "  [OK]\n\n"
  else
    printf "  [FAIL]\n\n"
    PUSH_FAILED=1
  fi
done

if [ "${PUSH_FAILED}" -eq 1 ]; then
  printf "*** Some pushes failed. Fix errors above before continuing. ***\n"
  exit 1
fi

# =============================================================================
# STEP 2: Pull catalog index from Artifactory and extract contents
# =============================================================================
printf "=== Step 2: Extracting catalog index from Artifactory ===\n\n"

mkdir -p "${WORKDIR}/catalog-raw"
mkdir -p "${WORKDIR}/catalog-content"

printf "  Pulling: %s\n" "${CATALOG_INDEX}"
skopeo copy --src-tls-verify=false \
  "docker://${CATALOG_INDEX}" \
  "dir:${WORKDIR}/catalog-raw"

# Extract layers to find dynamic-plugins.default.yaml and index.json
FOUND_CONTENT=0
for layer in "${WORKDIR}/catalog-raw"/*; do
  if [ -f "${layer}" ] && ! echo "${layer}" | grep -qE "(manifest\.json|version)$"; then
    if tar -xf "${layer}" -C "${WORKDIR}/catalog-content" 2>/dev/null; then
      if [ -f "${WORKDIR}/catalog-content/dynamic-plugins.default.yaml" ]; then
        FOUND_CONTENT=1
        printf "  [OK] Extracted catalog content from layer: %s\n\n" "$(basename "${layer}")"
        break
      fi
    fi
  fi
done

if [ "${FOUND_CONTENT}" -eq 0 ]; then
  printf "  [ERROR] Could not find dynamic-plugins.default.yaml in catalog index\n"
  exit 1
fi

# =============================================================================
# STEP 3: Rewrite ghcr.io references to Artifactory in both files
# =============================================================================
printf "=== Step 3: Rewriting OCI references ===\n\n"

DPDY="${WORKDIR}/catalog-content/dynamic-plugins.default.yaml"
INDEXJ="${WORKDIR}/catalog-content/index.json"

# Count ghcr.io references before
BEFORE_COUNT=$(grep -c "ghcr.io" "${DPDY}" 2>/dev/null || printf "0")
printf "  ghcr.io references in dynamic-plugins.default.yaml before: %s\n" "${BEFORE_COUNT}"

# Replace ghcr.io/redhat-developer/rhdh-plugin-export-overlays/ with Artifactory path
# This covers the OCI package paths in dynamic-plugins.default.yaml
sed -i "s|ghcr.io/redhat-developer/rhdh-plugin-export-overlays/|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/|g" "${DPDY}"

AFTER_COUNT=$(grep -c "ghcr.io" "${DPDY}" 2>/dev/null || printf "0")
printf "  ghcr.io references in dynamic-plugins.default.yaml after:  %s\n" "${AFTER_COUNT}"

# Also update index.json if it exists
if [ -f "${INDEXJ}" ]; then
  BEFORE_INDEX=$(grep -c "ghcr.io" "${INDEXJ}" 2>/dev/null || printf "0")
  printf "  ghcr.io references in index.json before: %s\n" "${BEFORE_INDEX}"

  # Use jq to update registryReference values
  if jq --arg target "${ARTIFACTORY}/${ARTIFACTORY_REPO}" '
    . | with_entries(
      .value.registryReference |= (
        if . and (. | test("ghcr.io")) then
          . | sub("ghcr.io/redhat-developer/rhdh-plugin-export-overlays/"; $target + "/rhdh-plugin-export-overlays/")
        else
          .
        end
      )
    )
  ' "${INDEXJ}" > "${INDEXJ}.tmp" 2>/dev/null; then
    mv "${INDEXJ}.tmp" "${INDEXJ}"
    AFTER_INDEX=$(grep -c "ghcr.io" "${INDEXJ}" 2>/dev/null || printf "0")
    printf "  ghcr.io references in index.json after:  %s\n" "${AFTER_INDEX}"
  else
    printf "  [WARN] jq update of index.json failed, continuing with sed\n"
    sed -i "s|ghcr.io/redhat-developer/rhdh-plugin-export-overlays/|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/|g" "${INDEXJ}"
  fi
fi

# Also update catalog-entities YAML files if present
YAML_COUNT=0
if [ -d "${WORKDIR}/catalog-content/catalog-entities" ]; then
  while IFS= read -r yaml_file; do
    if [ -n "${yaml_file}" ] && [ -f "${yaml_file}" ]; then
      sed -i "s|ghcr.io/redhat-developer/rhdh-plugin-export-overlays/|${ARTIFACTORY}/${ARTIFACTORY_REPO}/rhdh-plugin-export-overlays/|g" "${yaml_file}"
      YAML_COUNT=$((YAML_COUNT + 1))
    fi
  done < <(find "${WORKDIR}/catalog-content/catalog-entities" -name "*.yaml" -type f 2>/dev/null)
  printf "  Updated %s catalog-entity YAML files\n" "${YAML_COUNT}"
fi

printf "\n"

# =============================================================================
# STEP 4: Rebuild the catalog index image and push to Artifactory
# =============================================================================
printf "=== Step 4: Rebuilding and pushing catalog index ===\n\n"

BUILD_DIR="${WORKDIR}/build"
mkdir -p "${BUILD_DIR}/content"

# Copy all catalog content into build context
cp -r "${WORKDIR}/catalog-content"/* "${BUILD_DIR}/content/"

cat > "${BUILD_DIR}/Dockerfile" << 'EOF'
FROM scratch
COPY content/ /
EOF

TEMP_TAG="localhost/rhdh-catalog-index-rebuild:1.9"

printf "  Building image...\n"
if ! podman build -t "${TEMP_TAG}" "${BUILD_DIR}" 2>/dev/null; then
  printf "  [ERROR] podman build failed\n"
  exit 1
fi

printf "  Pushing to: %s\n" "${CATALOG_INDEX}"
if ! skopeo copy --all --dest-tls-verify=false \
  "containers-storage:${TEMP_TAG}" "docker://${CATALOG_INDEX}" 2>&1; then
  printf "  [ERROR] Failed to push rebuilt catalog index\n"
  podman rmi "${TEMP_TAG}" 2>/dev/null || true
  exit 1
fi

podman rmi "${TEMP_TAG}" 2>/dev/null || true
printf "  [OK] Catalog index rebuilt and pushed\n\n"

# =============================================================================
# STEP 5: Verify
# =============================================================================
printf "=== Step 5: Verification ===\n\n"

printf "  Catalog index in Artifactory:\n"
printf "    %s\n\n" "${CATALOG_INDEX}"

printf "  ArgoCD plugins in Artifactory:\n"
for entry in "${PLUGINS[@]}"; do
  target_image="${entry##*|}"
  name=$(echo "${target_image}" | sed 's|.*/||' | sed 's|:.*||')
  printf "    %s ... " "${name}"
  if skopeo inspect --tls-verify=false "docker://${target_image}" >/dev/null 2>&1; then
    printf "[OK]\n"
  else
    printf "[NOT FOUND]\n"
  fi
done

printf "\n"
printf "=== DONE ===\n\n"
printf "The catalog index at:\n"
printf "  %s\n" "${CATALOG_INDEX}"
printf "now has ghcr.io ArgoCD plugin references rewritten to Artifactory.\n\n"
printf "Your Backstage CR CATALOG_INDEX_IMAGE should point to:\n"
printf "  %s\n\n" "${CATALOG_INDEX}"
printf "In your dynamic-plugins-rhdh ConfigMap:\n"
printf "  - Keep includes: - dynamic-plugins.default.yaml\n"
printf "  - The 4 ArgoCD plugins are now pullable from Artifactory\n"
printf "  - You still need to DISABLE the remaining 34 ghcr.io plugins\n"
printf "    that you did NOT mirror (or mirror them all later)\n"
printf "  - Enable ArgoCD by adding disabled: false entries\n"
printf "    (the OCI paths in the catalog index already point to Artifactory)\n\n"

# Cleanup
rm -rf "${WORKDIR}"
