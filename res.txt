# ── 7. OPA policy: namespace required labels ──────────────────────────────────
cat > policies/opa/namespace.rego << 'EOF'
# policies/opa/namespace.rego
# Enforced in CI via conftest. Cluster-level enforcement via Kyverno.
package namespace

required_labels := {
  "company-xxx.com/map-id",
  "company-xxx.com/to-intake-number",
  "company-xxx.com/lob",
  "company-xxx.com/team",
  "company-xxx.com/environment",
  "app.kubernetes.io/managed-by"
}

deny[msg] {
  input.kind == "Namespace"
  label := required_labels[_]
  not input.metadata.labels[label]
  msg := sprintf("Namespace missing required label: %v", [label])
}

deny[msg] {
  input.kind == "Namespace"
  map_id := input.metadata.labels["company-xxx.com/map-id"]
  not regex.match(`^[0-9]{8}$`, map_id)
  msg := sprintf("company-xxx.com/map-id must be exactly 8 digits, got: %v", [map_id])
}

deny[msg] {
  input.kind == "Namespace"
  to_num := input.metadata.labels["company-xxx.com/to-intake-number"]
  not regex.match(`^TO-[0-9]{6}$`, to_num)
  msg := sprintf("company-xxx.com/to-intake-number must match TO-XXXXXX format, got: %v", [to_num])
}

deny[msg] {
  input.kind == "Namespace"
  managed_by := input.metadata.labels["app.kubernetes.io/managed-by"]
  managed_by != "gitops"
  msg := sprintf("app.kubernetes.io/managed-by must be 'gitops', got: %v", [managed_by])
}
EOF

# ── 8. OPA policy: machineset approved SKUs ──────────────────────────────────
cat > policies/opa/machineset.rego << 'EOF'
# policies/opa/machineset.rego
package machineset

approved_skus := {
  "Standard_D4s_v3",
  "Standard_D8s_v3",
  "Standard_D16s_v3",
  "Standard_E4s_v3",
  "Standard_E8s_v3"
}

deny[msg] {
  input.kind == "MachineSet"
  sku := input.spec.template.spec.providerSpec.value.vmSize
  not approved_skus[sku]
  msg := sprintf("MachineSet vmSize '%v' is not in the approved SKU list: %v", [sku, approved_skus])
}

deny[msg] {
  input.kind == "MachineSet"
  replicas := input.spec.replicas
  replicas > 10
  msg := sprintf("MachineSet replicas %v exceeds maximum of 10", [replicas])
}
EOF

# ── 9. Kyverno: require namespace labels (AUDIT MODE) ────────────────────────
cat > policies/kyverno/require-namespace-labels.yaml << 'EOF'
# policies/kyverno/require-namespace-labels.yaml
# AUDIT MODE: validationFailureAction: Audit
# Switch to Enforce after 2 weeks of audit-mode observation (Week 3)
---
apiVersion: kyverno.io/v1
kind: ClusterPolicy
metadata:
  name: require-namespace-labels
  annotations:
    policies.kyverno.io/title: Require Namespace Labels
    policies.kyverno.io/description: >
      All namespaces must carry company-xxx.com governance labels and
      app.kubernetes.io/managed-by=gitops for cost allocation and drift detection.
    platform.company-xxx.com/mode: audit
    platform.company-xxx.com/enforce-from: "TBD - switch after 2-week audit window"
spec:
  validationFailureAction: Audit
  background: true
  rules:
    - name: check-required-labels
      match:
        any:
          - resources:
              kinds:
                - Namespace
      validate:
        message: >
          Namespace must have labels: company-xxx.com/map-id,
          company-xxx.com/to-intake-number, company-xxx.com/lob,
          company-xxx.com/team, company-xxx.com/environment,
          app.kubernetes.io/managed-by=gitops
        pattern:
          metadata:
            labels:
              company-xxx.com/map-id: "?*"
              company-xxx.com/to-intake-number: "?*"
              company-xxx.com/lob: "?*"
              company-xxx.com/team: "?*"
              company-xxx.com/environment: "?*"
              app.kubernetes.io/managed-by: "gitops"
    - name: auto-inject-managed-by
      match:
        any:
          - resources:
              kinds:
                - Namespace
      mutate:
        patchStrategicMerge:
          metadata:
            labels:
              app.kubernetes.io/managed-by: "gitops"
EOF
