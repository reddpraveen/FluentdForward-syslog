#!/usr/bin/env bash
# =============================================================================
# setup-rhdh-and-gitops.sh
#
# COMPLETE SOLUTION - Single run creates:
#   1. All RHDH files (templates, custom action, catalog-info, app-config)
#   2. Updated bootstrap.sh (v5) with RHDH catalog-info generation added
#   3. Updated provision-resource.yml - workflow_dispatch block REMOVED
#      (RHDH templates are now the sole entry point for onboarding)
#   4. Updated render_rir.py - emits catalog-info.yaml per namespace
#   5. All OCP / ArgoCD / Kubernetes plugin config YAMLs for RHDH on hub cluster
#
# ARCHITECTURE ASSUMPTIONS (confirmed):
#   - RHDH is installed on the HUB cluster only (openshift-gitops namespace or
#     dedicated rhdh namespace)
#   - RHDH accesses TARGET clusters via the SAME ArgoCD cluster secrets already
#     present in the openshift-gitops namespace (label argocd.argoproj.io/secret-type=cluster)
#   - RHDH Kubernetes plugin reads those same secrets via a read-only SA
#   - GitHub integration uses GITOPS_BOT_TOKEN (same token as GHA)
#   - No Tekton, no AAP - GitHub Actions is sole CI/CD engine
#   - OCP OAuth is the identity provider for RHDH (SSO via existing OCP users/groups)
#
# VALIDATION LAYERS IMPLEMENTED:
#   Layer 1 (form-time)   : RHDH Kubernetes plugin queries cluster for NS uniqueness
#                           + AD group existence before user submits
#   Layer 2 (action-time) : patch-lob-input custom action schema + cluster registry check
#   Layer 3 (PR pipeline) : validate-pr.yml 7-stage (Step 2, existing placeholder)
#   Layer 4 (post-merge)  : RHDH catalog shows ArgoCD sync + live K8s resource state
#
# WHAT CHANGED FROM PREVIOUS SESSION:
#   - provision-resource.yml: workflow_dispatch trigger + patch-input job REMOVED
#     Only pull_request trigger remains (RHDH is now the entry point)
#   - render_rir.py: render_catalog_info() function added
#     Emits clusters/<cluster>/lob/<lob>/catalog/<ns>.catalog-info.yaml per namespace
#   - bootstrap.sh v5: adds rhdh/ directory skeleton + catalog dir per lob
#   - NEW: rhdh/app-config/app-config-rhdh.yaml (complete ConfigMap)
#   - NEW: rhdh/app-config/dynamic-plugins.yaml
#   - NEW: rhdh/ocp-manifests/ (SA, RBAC, Secret, Backstage CR)
#   - NEW: rhdh/catalog-info.yaml (registers templates + platform component)
#   - UPDATED: rhdh/templates/ (namespace + machineset - same as previous session)
#   - UPDATED: rhdh/actions/patch-lob-input/index.js (same as previous session)
#
# USAGE:
#   cd /path/to/ocp-gitops-platform   # your repo root
#   bash setup-rhdh-and-gitops.sh
#
# This script is IDEMPOTENT for the rhdh/ subtree - safe to re-run.
# For bootstrap.sh, render_rir.py, provision-resource.yml it OVERWRITES.
# =============================================================================

# ── 2a. ServiceAccount for RHDH Kubernetes plugin (read-only to target clusters)
cat > rhdh/ocp-manifests/rhdh-k8s-reader-sa.yaml << 'YAML'
# =============================================================================
# rhdh-k8s-reader-sa.yaml
#
# ServiceAccount used by RHDH Kubernetes plugin to query target clusters.
# RHDH runs on the HUB cluster. This SA is created on EACH TARGET cluster.
# The token is stored in a Secret in the RHDH namespace on the hub.
#
# Deploy to: ocp-nonprod-eastus-01 AND ocp-prod-eastus-01
# Command:
#   oc apply -f rhdh-k8s-reader-sa.yaml --context ocp-nonprod-eastus-01
#   oc apply -f rhdh-k8s-reader-sa.yaml --context ocp-prod-eastus-01
# =============================================================================
---
apiVersion: v1
kind: Namespace
metadata:
  name: rhdh-system
  labels:
    app.kubernetes.io/managed-by: gitops
    company-xxx.com/purpose: platform-tooling
---
apiVersion: v1
kind: ServiceAccount
metadata:
  name: rhdh-k8s-reader
  namespace: rhdh-system
  annotations:
    # Bound token projection (OCP 4.x - token auto-rotates)
    # Extract with: oc create token rhdh-k8s-reader -n rhdh-system --duration=8760h
    description: "Read-only SA for RHDH Kubernetes plugin - DO NOT grant write perms"
---
# ClusterRole: read namespaces, quotas, limitranges, netpols, rolebindings,
# machinesets, groups - everything RHDH needs to show on catalog component pages
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: rhdh-k8s-reader
  labels:
    app.kubernetes.io/managed-by: gitops
rules:
  # Core resources - what shows on Kubernetes tab in RHDH catalog
  - apiGroups: [""]
    resources:
      - namespaces
      - resourcequotas
      - limitranges
      - pods
      - events
      - serviceaccounts
    verbs: ["get", "list", "watch"]
  # Networking
  - apiGroups: ["networking.k8s.io"]
    resources: ["networkpolicies"]
    verbs: ["get", "list"]
  # RBAC - verify role bindings landed
  - apiGroups: ["rbac.authorization.k8s.io"]
    resources: ["rolebindings", "clusterrolebindings"]
    verbs: ["get", "list"]
  # MachineSet - for machineset onboarding validation
  - apiGroups: ["machine.openshift.io"]
    resources: ["machinesets", "machines"]
    verbs: ["get", "list"]
  # AD Groups - validate groups exist before RBAC onboarding
  - apiGroups: ["user.openshift.io"]
    resources: ["groups", "users"]
    verbs: ["get", "list"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: rhdh-k8s-reader
  labels:
    app.kubernetes.io/managed-by: gitops
subjects:
  - kind: ServiceAccount
    name: rhdh-k8s-reader
    namespace: rhdh-system
roleRef:
  apiGroup: rbac.authorization.k8s.io
  kind: ClusterRole
  name: rhdh-k8s-reader
YAML

# ── 2b. Secret in RHDH namespace on hub - holds tokens for both target clusters
cat > rhdh/ocp-manifests/rhdh-cluster-secrets.yaml << 'YAML'
# =============================================================================
# rhdh-cluster-secrets.yaml
#
# Secret holding SA tokens for RHDH Kubernetes plugin to reach target clusters.
# Deploy to: HUB cluster, rhdh namespace.
#
# HOW TO POPULATE:
#   # On each TARGET cluster, create a long-lived token:
#   oc create token rhdh-k8s-reader -n rhdh-system \
#     --duration=8760h \
#     --context ocp-nonprod-eastus-01
#
#   # Then base64-encode and paste below, OR use ESO/VSO to inject dynamically
#   # (recommended for production - rotate every 90 days)
#
# NOTE: ArgoCD already has cluster secrets in openshift-gitops namespace with
# label argocd.argoproj.io/secret-type=cluster. Those contain server URL and
# CA data you can reuse here. RHDH uses its own SA token, not the ArgoCD token.
# =============================================================================
---
apiVersion: v1
kind: Secret
metadata:
  name: rhdh-target-cluster-tokens
  namespace: rhdh            # Adjust to your RHDH namespace
  labels:
    app.kubernetes.io/managed-by: gitops
    app.kubernetes.io/component: rhdh-config
type: Opaque
stringData:
  # Tokens extracted from target clusters via: oc create token rhdh-k8s-reader
  # -n rhdh-system --duration=8760h
  OCP_NONPROD_SA_TOKEN: "<nonprod-rhdh-k8s-reader-token>"
  OCP_PROD_SA_TOKEN:    "<prod-rhdh-k8s-reader-token>"

  # API server URLs (same as what ArgoCD uses in its cluster secrets)
  OCP_NONPROD_API_URL:  "https://api.ocp-nonprod-eastus-01.example.com:6443"
  OCP_PROD_API_URL:     "https://api.ocp-prod-eastus-01.example.com:6443"

  # GitHub token - same as GHA GITOPS_BOT_TOKEN
  GITOPS_BOT_TOKEN: "<github-pat-contents-write-pr-write>"
YAML

# ── 2c. Backstage CR for RHDH Operator
cat > rhdh/ocp-manifests/backstage-cr.yaml << 'YAML'
# =============================================================================
# backstage-cr.yaml
#
# Backstage Custom Resource - drives RHDH Operator on the HUB cluster.
# Deploy to: HUB cluster, rhdh namespace.
#
# Assumes RHDH Operator is already installed via OperatorHub.
# =============================================================================
apiVersion: rhdh.redhat.com/v1alpha3
kind: Backstage
metadata:
  name: developer-hub
  namespace: rhdh
spec:
  application:
    # Main app config - references the ConfigMap created by app-config-rhdh.yaml
    appConfig:
      configMaps:
        - name: app-config-rhdh
          mountPath: /opt/app-root/src
    # Dynamic plugins config
    dynamicPluginsConfigMapName: dynamic-plugins-rhdh
    # Secrets injected as env vars
    extraEnvs:
      secrets:
        - name: rhdh-target-cluster-tokens
    replicas: 1
  route:
    enabled: true
  database:
    enableLocalDb: true    # Use built-in PostgreSQL; replace with external for HA
YAML

cat > rhdh/app-config/app-config-rhdh.yaml << 'YAML'
# =============================================================================
# app-config-rhdh.yaml
# ConfigMap: app-config-rhdh  Namespace: rhdh
#
# Complete RHDH app configuration for:
#   - GitHub integration (GITOPS_BOT_TOKEN)
#   - OpenShift OAuth (SSO via OCP identity provider)
#   - Kubernetes plugin (hub reads target clusters via SA tokens)
#   - ArgoCD plugin (shows GitOps sync status on catalog pages)
#   - Software Catalog (discovers templates from this repo)
#   - RBAC (platform-leads can run machineset template; all run namespace template)
#
# Deploy:  oc apply -f app-config-rhdh.yaml -n rhdh
# Reload:  oc rollout restart deployment/backstage-developer-hub -n rhdh
# =============================================================================
apiVersion: v1
kind: ConfigMap
metadata:
  name: app-config-rhdh
  namespace: rhdh
  labels:
    app.kubernetes.io/managed-by: gitops
    app.kubernetes.io/component: rhdh-config
data:
  app-config-rhdh.yaml: |
    app:
      title: "Platform Developer Hub"
      baseUrl: https://backstage-developer-hub-rhdh.apps.ocp-hub.example.com  # Update to your route

    backend:
      baseUrl: https://backstage-developer-hub-rhdh.apps.ocp-hub.example.com
      cors:
        origin: https://backstage-developer-hub-rhdh.apps.ocp-hub.example.com
      # PostgreSQL (managed by Operator when enableLocalDb: true)
      database:
        client: pg
        connection:
          host: ${POSTGRES_HOST}
          port: ${POSTGRES_PORT}
          user: ${POSTGRES_USER}
          password: ${POSTGRES_PASSWORD}

    # ── GitHub Integration ──────────────────────────────────────────────────
    # Uses GITOPS_BOT_TOKEN (contents:write + pull-requests:write)
    # Same token as GHA GITOPS_BOT_TOKEN secret - no separate token needed
    integrations:
      github:
        - host: github.com                   # Replace with your GHES hostname if applicable
          token: ${GITOPS_BOT_TOKEN}

    # ── Authentication: OpenShift OAuth ────────────────────────────────────
    # Users authenticate with their OCP credentials (AD-backed via OCP OAuth)
    # This means AD group membership is automatically available in RHDH
    auth:
      environment: production
      providers:
        github:
          production:
            clientId: ${GITHUB_CLIENT_ID}      # GitHub OAuth App client ID
            clientSecret: ${GITHUB_CLIENT_SECRET}
        # Alternatively use OCP OAuth directly:
        # oidc:
        #   production:
        #     metadataUrl: https://api.ocp-hub.example.com:6443/.well-known/oauth-authorization-server
        #     clientId: ${OCP_OAUTH_CLIENT_ID}
        #     clientSecret: ${OCP_OAUTH_CLIENT_SECRET}
        #     scope: 'openid profile email groups'

    # ── Software Catalog ───────────────────────────────────────────────────
    catalog:
      # Allow these entity kinds from our repo
      rules:
        - allow: [Component, System, API, Resource, Location, Template, Group, User]

      # Discover catalog entities from the GitOps platform repo
      # The root catalog-info.yaml uses glob Location to pull in all per-namespace
      # catalog-info.yaml files generated by render_rir.py
      locations:
        - type: url
          target: https://github.com/org-xxx/ocp-gitops-platform/blob/main/rhdh/catalog-info.yaml
          rules:
            - allow: [Component, Template, Location, System]

      # Org data: sync GitHub teams as Groups into the catalog
      # This makes 'group:default/platform-leads' etc available for RBAC
      providers:
        github:
          orgProvider:
            id: org-xxx
            githubUrl: https://github.com/org-xxx
            schedule:
              frequency: { minutes: 30 }
              timeout:   { minutes: 5 }

    # ── Kubernetes Plugin ──────────────────────────────────────────────────
    # RHDH on HUB cluster reads TARGET clusters using dedicated SA tokens.
    # Cluster names here MUST match backstage.io/kubernetes-cluster annotations
    # in catalog-info.yaml files generated by render_rir.py.
    #
    # Token source: rhdh-k8s-reader SA on each target cluster
    # (created by rhdh/ocp-manifests/rhdh-k8s-reader-sa.yaml)
    kubernetes:
      serviceLocatorMethod:
        type: multiTenant
      clusterLocatorMethods:
        - type: config
          clusters:
            - name: ocp-nonprod-eastus-01
              url:  ${OCP_NONPROD_API_URL}
              authProvider: serviceAccount
              serviceAccountToken: ${OCP_NONPROD_SA_TOKEN}
              skipTLSVerify: false
              # caData: base64 of cluster CA cert (get from ArgoCD cluster secret)
              # oc get secret <argocd-cluster-secret> -n openshift-gitops \
              #   -o jsonpath='{.data.config}' | base64 -d | jq -r '.tlsClientConfig.caData'

            - name: ocp-prod-eastus-01
              url:  ${OCP_PROD_API_URL}
              authProvider: serviceAccount
              serviceAccountToken: ${OCP_PROD_SA_TOKEN}
              skipTLSVerify: false

    # ── ArgoCD Plugin ──────────────────────────────────────────────────────
    # Connects to the ArgoCD instance on the HUB cluster (openshift-gitops).
    # Shows sync status + health on every catalog component page that has
    # the argocd/app-selector annotation.
    argocd:
      username: ${ARGOCD_USER}
      password: ${ARGOCD_PASSWORD}
      waitCycles: 25
      appLocatorMethods:
        - type: config
          instances:
            - name: hub-argocd
              url: ${ARGOCD_URL}    # oc get route openshift-gitops-server -n openshift-gitops

    # ── Scaffolder ─────────────────────────────────────────────────────────
    scaffolder:
      # GitHub integration for PR creation by RHDH scaffolder
      # Uses same GITOPS_BOT_TOKEN
      github:
        token: ${GITOPS_BOT_TOKEN}

    # ── RBAC ───────────────────────────────────────────────────────────────
    permission:
      enabled: true
      rbac:
        admin:
          users:
            - name: group:default/platform-leads
        policies-csv-file: /opt/app-root/src/rbac-policy.csv
        policyFileReload: true

    # ── Proxy ──────────────────────────────────────────────────────────────
    # Required for ArgoCD plugin to call ArgoCD API from browser
    proxy:
      endpoints:
        /argocd/api:
          target: ${ARGOCD_URL}/api/v1/
          changeOrigin: true
          # Secure token header - RHDH passes this to ArgoCD
          headers:
            Cookie: argocd.token=${ARGOCD_TOKEN}

---
# RBAC policy file - mounted alongside app-config
apiVersion: v1
kind: ConfigMap
metadata:
  name: rbac-policy-rhdh
  namespace: rhdh
  labels:
    app.kubernetes.io/managed-by: gitops
data:
  rbac-policy.csv: |
    # Platform leads: admin rights + can run machineset template
    p, role:default/platform-leads, catalog-entity,           read,    allow
    p, role:default/platform-leads, catalog.entity.create,    create,  allow
    p, role:default/platform-leads, scaffolder-template,      execute, allow
    p, role:default/platform-leads, scaffolder-action,        use,     allow
    g, group:default/platform-leads, role:default/platform-leads

    # All authenticated users: can read catalog + run namespace onboarding
    p, role:default/authenticated, catalog-entity,          read,   allow
    p, role:default/authenticated, catalog.entity.create,   create, allow
    # Namespace template only - machineset template is blocked for non-platform-leads
    p, role:default/authenticated, scaffolder-template,     execute, allow
    p, role:default/authenticated, scaffolder-action,       use,     allow
    g, group:default/all-users, role:default/authenticated

    # Explicit deny: non-platform-leads cannot execute machineset template
    p, role:default/authenticated, scaffolder-template:default/machineset-onboarding, execute, deny
YAML

/**
 * patch-lob-input  (v1.1 — consolidated)
 * ----------------------------------------
 * Custom Backstage scaffolder action.
 * Fetches onboarding/<lob>.input.yaml from GitHub, merges the patch entry,
 * returns merged YAML for the scaffolder to commit via pull-request.
 *
 * patchType: 'namespace' | 'machineset'
 *
 * Merge rules:
 *   Scalar fields     : patch wins
 *   governance        : deep merge, null removes key
 *   roles list        : keyed by group; null role removes group
 *   labels map        : additive; null removes key
 *   machineSet block  : full replace when provided
 */
const { createTemplateAction } = require('@backstage/plugin-scaffolder-node');
const { Octokit } = require('@octokit/rest');
const yaml = require('js-yaml');

const VALID_LOBS      = ['lob1','lob2','lob3','lob4','lob5'];
const VALID_ENVS      = ['dev','sit','uat','prod'];
const VALID_QUOTAS    = ['small','medium','large'];
const VALID_NETPOLS   = ['deny-all','allow-same-namespace','lob-internal'];
const VALID_DC        = ['public','internal','confidential','restricted'];
const APPROVED_SKUS   = ['Standard_D4s_v3','Standard_D8s_v3','Standard_D16s_v3','Standard_E4s_v3','Standard_E8s_v3'];

function mergeRoles(existing = [], patch = []) {
  const map = Object.fromEntries(existing.map(r => [r.group, r]));
  for (const e of patch) {
    if (!e.group) continue;
    if (e.role === null) delete map[e.group]; else map[e.group] = e;
  }
  return Object.values(map);
}

function mergeLabels(existing = {}, patch = {}) {
  const r = { ...existing };
  for (const [k,v] of Object.entries(patch)) { if (v === null) delete r[k]; else r[k] = v; }
  return r;
}

function mergeGovernance(existing = {}, patch = {}) {
  const r = { ...existing };
  for (const [k,v] of Object.entries(patch)) {
    if (v === null || v === '') delete r[k]; else if (v !== undefined) r[k] = v;
  }
  return r;
}

function validateNamespacePatch(entry, lob) {
  const e = [];
  if (!/^[a-z][a-z0-9-]{2,62}$/.test(entry.namespace || '')) e.push('namespace pattern invalid');
  if (!VALID_LOBS.includes(lob)) e.push('invalid lob');
  if (!VALID_ENVS.includes(entry.environment)) e.push('invalid environment');
  const g = entry.governance || {};
  if (!/^[0-9]{8}$/.test(g['map-id'] || '')) e.push('governance.map-id must be 8 digits');
  if (!/^TO-[0-9]{6}$/.test(g['to-intake'] || '')) e.push('governance.to-intake must match TO-XXXXXX');
  if (!/^CHG-[A-Z0-9]+$/.test(g['change-ref'] || '')) e.push('governance.change-ref must match CHG-XXXXXXX');
  if (!g['team-name']) e.push('governance.team-name required');
  if (entry.quota && !VALID_QUOTAS.includes(entry.quota)) e.push('invalid quota');
  if (entry.netpol && !VALID_NETPOLS.includes(entry.netpol)) e.push('invalid netpol');
  if (g['data-classification'] && !VALID_DC.includes(g['data-classification'])) e.push('invalid data-classification');
  return e;
}

function validateMachinesetPatch(entry) {
  const e = [];
  if (!/^[a-z][a-z0-9-]{2,62}$/.test(entry.namespace || '')) e.push('namespace pattern invalid');
  if (!/^CHG-[A-Z0-9]+$/.test((entry.governance || {})['change-ref'] || '')) e.push('governance.change-ref invalid');
  const ms = entry.machineSet || {};
  if (!APPROVED_SKUS.includes(ms.instanceType)) e.push(`instanceType must be one of: ${APPROVED_SKUS.join(', ')}`);
  const rep = Number(ms.replicas);
  if (!Number.isInteger(rep) || rep < 1 || rep > 10) e.push('replicas must be 1-10');
  if (!['1','2','3'].includes(String(ms.zone))) e.push('zone must be 1, 2, or 3');
  return e;
}

function mergeEntry(existingNamespaces, patch, patchType) {
  const namespaces = existingNamespaces ? [...existingNamespaces] : [];
  const idx = namespaces.findIndex(n => n.namespace === patch.namespace);
  const existing = idx >= 0 ? namespaces[idx] : {};
  let merged = { ...existing };

  if (patchType === 'namespace') {
    merged.namespace   = patch.namespace;
    merged.cluster     = patch.cluster     || existing.cluster;
    merged.environment = patch.environment || existing.environment;
    if (patch.quota)  merged.quota  = patch.quota;
    if (patch.netpol) merged.netpol = patch.netpol;
    merged.governance = mergeGovernance(existing.governance, patch.governance);
    if (patch.roles  && patch.roles.length  > 0) merged.roles  = mergeRoles(existing.roles,  patch.roles);
    if (patch.labels && Object.keys(patch.labels).length > 0) merged.labels = mergeLabels(existing.labels, patch.labels);
  } else {
    merged.governance = mergeGovernance(existing.governance, { 'change-ref': (patch.governance||{})['change-ref'] });
    if (patch.machineSet) merged.machineSet = patch.machineSet;
  }

  if (idx >= 0) namespaces[idx] = merged; else namespaces.push(merged);
  return namespaces;
}

const patchLobInputAction = createTemplateAction({
  id: 'platform:patch-lob-input',
  description: 'Fetch onboarding/<lob>.input.yaml, merge patch entry, return merged YAML',
  schema: {
    input: {
      type: 'object',
      required: ['repoOwner','repoName','token','lob','patchType','entry'],
      properties: {
        repoOwner:  { type: 'string' },
        repoName:   { type: 'string' },
        token:      { type: 'string' },
        lob:        { type: 'string' },
        patchType:  { type: 'string', enum: ['namespace','machineset'] },
        entry:      { type: 'object' },
        baseBranch: { type: 'string' },
      },
    },
    output: {
      type: 'object',
      properties: {
        mergedYaml: { type: 'string' },
        filePath:   { type: 'string' },
        isNew:      { type: 'boolean' },
      },
    },
  },
  async handler(ctx) {
    const { repoOwner, repoName, token, lob, patchType, entry, baseBranch = 'main' } = ctx.input;
    ctx.logger.info(`patch-lob-input: lob=${lob} patchType=${patchType} ns=${entry.namespace}`);

    const errors = patchType === 'namespace'
      ? validateNamespacePatch(entry, lob)
      : validateMachinesetPatch(entry);
    if (errors.length) throw new Error(`Validation failed:\n${errors.map(e=>`  - ${e}`).join('\n')}`);

    const octokit = new Octokit({ auth: token });
    const filePath = `onboarding/${lob}.input.yaml`;
    let existingContent = null, existingSha = null;

    try {
      const { data } = await octokit.repos.getContent({ owner: repoOwner, repo: repoName, path: filePath, ref: baseBranch });
      existingContent = Buffer.from(data.content,'base64').toString('utf-8');
      existingSha = data.sha;
    } catch(err) {
      if (err.status !== 404) throw err;
    }

    const existingData = existingContent ? yaml.load(existingContent) : {};
    const existingNamespaces = existingData.namespaces || [];
    const existingEntry = existingNamespaces.find(n => n.namespace === entry.namespace);

    if (patchType === 'machineset' && !existingEntry)
      throw new Error(`Namespace '${entry.namespace}' not found in ${filePath}. Onboard namespace first.`);

    const isNew = !existingEntry;
    const mergedNamespaces = mergeEntry(existingNamespaces, entry, patchType);
    const mergedData = { ...existingData, namespaces: mergedNamespaces };

    const header = [
      `# onboarding/${lob}.input.yaml`,
      `# Managed by RHDH Software Templates - do not edit structure manually`,
      `# Last patched: ${new Date().toISOString()}`,
      ``,
    ].join('\n');

    const mergedYaml = header + yaml.dump(mergedData, { lineWidth: 120, noRefs: true, sortKeys: false });

    ctx.output('mergedYaml', mergedYaml);
    ctx.output('filePath',   filePath);
    ctx.output('isNew',      isNew);
  },
});

/**
 * validate-onboarding
 * --------------------
 * Custom scaffolder action for Layer 1 validation.
 * Queries the live OCP cluster (via RHDH Kubernetes plugin backend)
 * BEFORE the PR is opened to catch errors early.
 *
 * Checks:
 *   1. Namespace does not already exist on the target cluster
 *      (catches drift between Git state and cluster state)
 *   2. AD groups specified for RBAC actually exist in OCP
 *      (prevents RBAC bindings that will never resolve)
 *
 * This action runs as a scaffolder step - it uses the Kubernetes
 * plugin's cluster config (same SA tokens from app-config-rhdh.yaml)
 * via the backend Kubernetes client, NOT the user's browser token.
 *
 * Registration: same scaffolder registration as patch-lob-input.
 */
const { createTemplateAction } = require('@backstage/plugin-scaffolder-node');

const validateOnboardingAction = createTemplateAction({
  id: 'platform:validate-onboarding',
  description: 'Validate namespace uniqueness and AD group existence against live OCP cluster',
  schema: {
    input: {
      type: 'object',
      required: ['clusterName', 'namespace'],
      properties: {
        clusterName:  { type: 'string', description: 'Cluster name matching kubernetes config' },
        namespace:    { type: 'string', description: 'Namespace to check for uniqueness' },
        adminGroup:   { type: 'string', description: 'OCP AD group name for admin role (optional)' },
        editGroup:    { type: 'string', description: 'OCP AD group name for edit role (optional)' },
        viewGroup:    { type: 'string', description: 'OCP AD group name for view role (optional)' },
      },
    },
    output: {
      type: 'object',
      properties: {
        namespaceExists: { type: 'boolean' },
        groupsValid:     { type: 'boolean' },
        warnings:        { type: 'array', items: { type: 'string' } },
      },
    },
  },

  async handler(ctx) {
    const { clusterName, namespace, adminGroup, editGroup, viewGroup } = ctx.input;
    const warnings = [];

    ctx.logger.info(`validate-onboarding: cluster=${clusterName} namespace=${namespace}`);

    // Access the Kubernetes client via the scaffolder context
    // The Kubernetes plugin registers its client in the backend service locator
    // ctx.getClient('kubernetes') provides cluster-aware HTTP client
    const kubeClient = ctx.getClient ? ctx.getClient('kubernetes') : null;

    if (!kubeClient) {
      // Graceful degradation: if Kubernetes client is not available (e.g. local dev),
      // warn but don't block. The GHA pipeline (validate-pr.yml) is the hard gate.
      ctx.logger.warn('Kubernetes client not available - skipping live cluster validation');
      warnings.push('Live cluster validation skipped (Kubernetes plugin not configured)');
      ctx.output('namespaceExists', false);
      ctx.output('groupsValid',     true);
      ctx.output('warnings',        warnings);
      return;
    }

    // ── Check 1: Namespace uniqueness ──────────────────────────────────────
    try {
      const nsResponse = await kubeClient.fetch({
        clusterName,
        path: `/api/v1/namespaces/${namespace}`,
      });
      if (nsResponse.status === 200) {
        // Namespace exists - could be an update (allowed) or collision (block for new)
        ctx.logger.warn(`Namespace '${namespace}' already exists on ${clusterName}`);
        warnings.push(`Namespace '${namespace}' already exists on ${clusterName}. ` +
          `If this is an update request, the provisioning pipeline will handle it. ` +
          `If this is a new request, verify you are not colliding with an existing namespace.`);
        ctx.output('namespaceExists', true);
      } else {
        ctx.output('namespaceExists', false);

        ctx.output('namespaceExists', false);
      } else {
        ctx.logger.warn(`Could not check namespace existence: ${err.message}`);
        warnings.push(`Could not verify namespace uniqueness: ${err.message}`);
        ctx.output('namespaceExists', false);
      }
    }

    // ── Check 2: AD Group existence ─────────────────────────────────────────
    const groupsToCheck = [adminGroup, editGroup, viewGroup].filter(Boolean);
    const missingGroups = [];

    for (const group of groupsToCheck) {
      try {
        const groupResponse = await kubeClient.fetch({
          clusterName,
          path: `/apis/user.openshift.io/v1/groups/${encodeURIComponent(group)}`,
        });
        if (groupResponse.status !== 200) {
          missingGroups.push(group);
          ctx.logger.warn(`AD group '${group}' not found on ${clusterName}`);
        } else {
          ctx.logger.info(`AD group '${group}' found on ${clusterName}`);
        }
      } catch (err) {
        if (err.status === 404) {
          missingGroups.push(group);
        } else {
          ctx.logger.warn(`Could not check group '${group}': ${err.message}`);
          warnings.push(`Could not verify group '${group}': ${err.message}`);
        }
      }

      // Hard failure - AD groups that don't exist will create orphaned RoleBindings
      throw new Error(
        `The following AD groups do not exist on cluster '${clusterName}':\n` +
        missingGroups.map(g => `  - ${g}`).join('\n') + '\n\n' +
        `Verify the group names with your Active Directory admin and ensure ` +
        `OCP group sync has run (oc get groups | grep <groupname>).`
      );
    }

    ctx.output('groupsValid', true);
    ctx.output('warnings',    warnings);

    if (warnings.length > 0) {
      ctx.logger.warn(`Validation completed with ${warnings.length} warning(s)`);
    } else {
      ctx.logger.info('Validation passed - all checks OK');

def render_catalog_info(ns_entry: dict, cluster: str) -> str:
    """
    Generates a Backstage catalog-info.yaml for a provisioned namespace.

    The generated file is placed at:
      clusters/<cluster>/lob/<lob>/catalog/<namespace>.catalog-info.yaml

    It is discovered by RHDH via the glob Location in rhdh/namespace-locations.yaml.

    Annotations set:
      backstage.io/kubernetes-id        -> links to Kubernetes plugin tab
      backstage.io/kubernetes-namespace -> scopes K8s plugin to this namespace
      backstage.io/kubernetes-cluster   -> which cluster to query
      argocd/app-selector               -> links to ArgoCD sync status widget
      github.com/project-slug           -> links to GitHub repo in RHDH
    """
    namespace  = ns_entry["namespace"]
    lob        = ns_entry.get("lob", "unknown")
    environment = ns_entry.get("environment", "unknown")
    governance  = ns_entry.get("governance", {})
    team_name   = governance.get("team-name", f"{lob}-team")
    map_id      = governance.get("map-id", "")
    change_ref  = governance.get("change-ref", "")

    # Owner group: use LOB leads group convention
    owner_group = f"group:default/{lob}-leads"

    catalog_entity = {
        "apiVersion": "backstage.io/v1alpha1",
        "kind": "Component",
        "metadata": {
            "name": namespace,
            "title": f"{namespace} ({environment})",
            "description": (
                f"OpenShift namespace provisioned via GitOps platform. "
                f"LOB: {lob} | Team: {team_name} | MAP: {map_id}"
            ),
            "annotations": {
                # Kubernetes plugin - shows live NS resources on catalog page
                "backstage.io/kubernetes-id":        namespace,
                "backstage.io/kubernetes-namespace": namespace,
                "backstage.io/kubernetes-cluster":   cluster,

                # ArgoCD plugin - shows sync/health on overview card
                # Selector matches the labels applied to the Argo CD Application
                # generated by argocd/appsets/ (Step 3)
                "argocd/app-selector": (
                    f"company-xxx.com/lob={lob},"
                    f"company-xxx.com/namespace={namespace}"
                ),

                # GitHub - links back to the repo in RHDH UI
                "github.com/project-slug": "org-xxx/ocp-gitops-platform",

                # Traceability
                "company-xxx.com/map-id":     map_id,
                "company-xxx.com/change-ref": change_ref,
                "company-xxx.com/lob":        lob,
                "company-xxx.com/environment": environment,
            },
            "tags": [lob, environment, "openshift", "namespace"],
            "links": [
                {
                    "url": f"https://console-openshift-console.apps.{cluster}.example.com/k8s/cluster/namespaces/{namespace}",
                    "title": "OpenShift Console",
                    "icon": "web",
                },

        },
        "spec": {
            "type": "namespace",
            "lifecycle": "production" if environment == "prod" else "experimental",
            "owner": owner_group,
            "system": "ocp-gitops-platform",
        },

    header = (
        f"# {namespace}.catalog-info.yaml\n"
        f"# Auto-generated by render_rir.py - DO NOT EDIT MANUALLY\n"
        f"# Generated: {datetime.now(timezone.utc).isoformat()}\n"
        f"# Cluster: {cluster}  LOB: {lob}  Environment: {environment}\n"
        f"# Discovered by RHDH via rhdh/namespace-locations.yaml glob Location\n\n"
    )

    return header + yaml.dump(catalog_entity, default_flow_style=False, sort_keys=False)

# Template rendering

def render_template(env: Environment, template_name: str, context: dict) -> str:
    template = env.get_template(template_name)
    return template.render(**context)


def build_render_context(ns_entry: dict, resource_type: str) -> dict:
    namespace    = ns_entry["namespace"]
    cluster      = ns_entry.get("cluster", "")
    environment  = ns_entry.get("environment", "dev")
    governance   = ns_entry.get("governance", {})
    quota_name   = ns_entry.get("quota", "small")
    netpol_name  = ns_entry.get("netpol", "deny-all")
    roles        = ns_entry.get("roles", [])
    labels_extra = ns_entry.get("labels", {})
    machine_set  = ns_entry.get("machineSet", {})

    quota_values = QUOTA_PROFILES.get(quota_name, QUOTA_PROFILES["small"])
    lr_profile   = "relaxed" if quota_name == "large" else "standard"
    lr_values    = LIMITRANGE_PROFILES[lr_profile]

    # Standard labels applied to all generated resources
    standard_labels = {
        "app.kubernetes.io/managed-by":  "gitops",
        "company-xxx.com/lob":           ns_entry.get("lob", ""),
        "company-xxx.com/environment":   environment,
        "company-xxx.com/map-id":        governance.get("map-id", ""),
        "company-xxx.com/to-intake-number": governance.get("to-intake", ""),
        "company-xxx.com/team":          governance.get("team-name", ""),
        "company-xxx.com/namespace":     namespace,
    }
    standard_labels.update(labels_extra)

    return {
        "namespace":        namespace,
        "cluster":          cluster,
        "environment":      environment,
        "governance":       governance,
        "quota_name":       quota_name,
        "netpol_name":      netpol_name,
        "roles":            roles,
        "labels":           standard_labels,
        "machine_set":      machine_set,
        "quota":            quota_values,
        "limitrange":       lr_values,
        "lr_profile":       lr_profile,
        "generated_at":     datetime.now(timezone.utc).isoformat(),
    }


def write_file(path: str, content: str) -> None:
    os.makedirs(os.path.dirname(path), exist_ok=True)
    with open(path, "w") as f:
        f.write(content)
    print(f"  Written: {path}")


# ---------------------------------------------------------------------------
# Main
# ---------------------------------------------------------------------------
def main():
    parser = argparse.ArgumentParser(description="Render GitOps cluster YAMLs from RIR manifest")
    parser.add_argument("--manifest",  required=True, help="Path to render manifest JSON (from input_to_rir.py)")
    parser.add_argument("--templates", required=True, help="Path to Jinja2 templates directory")
    parser.add_argument("--outdir",    required=True, help="Root output directory (clusters/)")
    args = parser.parse_args()

    if not os.path.exists(args.manifest):
        print(f"ERROR: Manifest not found: {args.manifest}", file=sys.stderr)
        sys.exit(1)

    with open(args.manifest) as f:
        manifest = json.load(f)

    jinja_env = Environment(
        loader=FileSystemLoader(args.templates),
        undefined=StrictUndefined,
        trim_blocks=True,
        lstrip_blocks=True,
    )

    rendered_count = 0
    errors = []

    for item in manifest.get("namespaces", []):
        ns_entry      = item["entry"]
        cluster       = item["cluster"]
        lob           = item["lob"]
        changed_types = item.get("changed_resource_types", [])
        namespace     = ns_entry["namespace"]

        base_path = os.path.join(args.outdir, cluster, "lob", lob)

        print(f"\nRendering {namespace} on {cluster} (changed: {changed_types})")

        for resource_type in changed_types:
            template_name = RESOURCE_TYPE_TO_TEMPLATE.get(resource_type)
            if not template_name:
                print(f"  WARNING: No template for resource type '{resource_type}' - skipping")
                continue

            subdir = resource_type + "s" if not resource_type.endswith("s") else resource_type
            if resource_type == "netpol":
                subdir = "netpol"
            elif resource_type == "machineset":
                subdir = "machinesets"
            elif resource_type == "rbac":
                subdir = "rbac"
            elif resource_type == "quota":
                subdir = "quotas"
            elif resource_type == "limitrange":
                subdir = "limitranges"
            elif resource_type == "namespace":
                subdir = "namespaces"

            out_path = os.path.join(base_path, subdir, f"{namespace}.yaml")

            try:
                context = build_render_context(ns_entry, resource_type)
                content = render_template(jinja_env, template_name, context)
                write_file(out_path, content)
                rendered_count += 1
            except Exception as e:
                err_msg = f"Failed to render {resource_type} for {namespace}: {e}"
                print(f"  ERROR: {err_msg}", file=sys.stderr)
                errors.append(err_msg)

        # ── NEW: Always render catalog-info.yaml for changed namespaces ───
        # This is rendered regardless of which resource types changed.
        # Even a governance-only update should refresh the catalog entry.
        catalog_path = os.path.join(base_path, "catalog", f"{namespace}.catalog-info.yaml")
        try:
            catalog_content = render_catalog_info(ns_entry, cluster)
            write_file(catalog_path, catalog_content)
            rendered_count += 1
        except Exception as e:
            err_msg = f"Failed to render catalog-info for {namespace}: {e}"
            print(f"  ERROR: {err_msg}", file=sys.stderr)
            errors.append(err_msg)

    print(f"\nRendering complete. Files written: {rendered_count}")

    if errors:
        print(f"\nERRORS ({len(errors)}):", file=sys.stderr)
        for err in errors:
            print(f"  - {err}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
