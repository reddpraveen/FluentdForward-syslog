#!/usr/bin/env python3
"""
input_to_rir.py
---------------
Reads onboarding/<lob>.input.yaml (multi-namespace LOB file).
For each namespace entry:
  1. Validates cluster name against clusters/cluster-registry.yaml  (#10)
  2. Checks for duplicate namespace across other LOBs               (#6)
  3. Merges with existing RIR using deep merge + null-removal
  4. Applies defaults (quota: small, netpol: deny-all,
     data-classification: internal)                                  (#14)
  5. Validates merged state against input schema
  6. Writes resource-catalog/<lob>/<namespace>.rir.yaml
  7. Writes resource-catalog/<lob>/<namespace>.audit.yaml

Emits a JSON render manifest consumed by render_rir.py.

Usage:
  python3 scripts/input_to_rir.py \
    --input    onboarding/lob1.input.yaml \
    --schema   resource-catalog/schemas/rir-v1-schema.json \
    --registry clusters/cluster-registry.yaml \
    --outdir   resource-catalog \
    --author   platform-engineer \
    --manifest /tmp/render-manifest.json
"""

import argparse
import copy
import json
import os
import re
import sys
from datetime import datetime, timezone

import yaml


# ---------------------------------------------------------------------------
# Constants
# ---------------------------------------------------------------------------
DEFAULTS = {
    "quota":               "small",
    "netpol":              "deny-all",
    "data-classification": "internal",    # #14
}

QUOTA_PROFILES = {
    "small":  {"requests.cpu": "2",  "requests.memory": "4Gi",  "limits.cpu": "4",  "limits.memory": "8Gi",  "pods": "20"},
    "medium": {"requests.cpu": "8",  "requests.memory": "16Gi", "limits.cpu": "16", "limits.memory": "32Gi", "pods": "50"},
    "large":  {"requests.cpu": "16", "requests.memory": "32Gi", "limits.cpu": "32", "limits.memory": "64Gi", "pods": "100"},
}

LIMITRANGE_FROM_QUOTA = {
    "small":  "standard",
    "medium": "standard",
    "large":  "relaxed",
}

ROLE_MAP = {
    "admin":  "admin",
    "edit":   "edit",
    "view":   "view",
    "reader": "cluster-reader",
}

APPROVED_SKUS = {
    "Standard_D4s_v3", "Standard_D8s_v3", "Standard_D16s_v3",
    "Standard_E4s_v3", "Standard_E8s_v3",
}

VALID_LOBS         = {"lob1", "lob2", "lob3", "lob4", "lob5"}
VALID_ENVIRONMENTS = {"dev", "sit", "uat", "prod"}
VALID_NETPOLS      = {"deny-all", "allow-same-namespace", "lob-internal"}
VALID_QUOTAS       = set(QUOTA_PROFILES.keys())
VALID_DATA_CLASS   = {"public", "internal", "confidential", "restricted"}


# ---------------------------------------------------------------------------
# Load cluster registry (#10)
# Returns set of active cluster names, or None if registry not found
# ---------------------------------------------------------------------------
def load_cluster_registry(registry_path: str) -> set | None:
    if not os.path.exists(registry_path):
        print(f"WARNING: cluster-registry.yaml not found at {registry_path} - skipping cluster validation")
        return None
    with open(registry_path) as f:
        data = yaml.safe_load(f)
    return {c["name"] for c in data.get("clusters", []) if c.get("status") == "active"}


# ---------------------------------------------------------------------------
# Cross-LOB duplicate namespace detection (#6)
# ---------------------------------------------------------------------------
def check_duplicate_namespace(ns_name: str, lob: str, catalog_dir: str) -> list:
    """Returns list of other LOBs that already have a RIR for this namespace."""
    duplicates = []
    if not os.path.isdir(catalog_dir):
        return duplicates
    for lob_dir in os.listdir(catalog_dir):
        if lob_dir == lob:
            continue
        lob_path = os.path.join(catalog_dir, lob_dir)
        if not os.path.isdir(lob_path):
            continue
        rir_path = os.path.join(lob_path, f"{ns_name}.rir.yaml")
        if os.path.exists(rir_path):
            duplicates.append(lob_dir)
    return duplicates


# ---------------------------------------------------------------------------
# Merge helpers
# ---------------------------------------------------------------------------
def merge_roles(existing: list, patch: list) -> list:
    """
    Merge roles keyed by group name.
    New groups appended, existing preserved, null role removes group.
    """
    role_map = {r["group"]: r for r in (existing or [])}
    for entry in (patch or []):
        group = entry.get("group")
        if not group:
            continue
        if entry.get("role") is None:
            role_map.pop(group, None)
        else:
            role_map[group] = entry
    return list(role_map.values())


def merge_labels(existing: dict, patch: dict) -> dict:
    """Additive label merge. Null value removes the label."""
    result = copy.deepcopy(existing or {})
    for k, v in (patch or {}).items():
        if v is None:
            result.pop(k, None)
        else:
            result[k] = v
    return result


# ---------------------------------------------------------------------------
# Validation
# ---------------------------------------------------------------------------
def validate_entry(entry: dict, lob: str, valid_clusters: set | None) -> list:
    errors = []

    ns = entry.get("namespace", "")
    if not re.match(r"^[a-z][a-z0-9-]{2,62}$", str(ns)):
        errors.append(f"namespace '{ns}' must match ^[a-z][a-z0-9-]{{2,62}}$")

    cluster = entry.get("cluster", "")
    if valid_clusters is not None and cluster not in valid_clusters:
        errors.append(f"cluster '{cluster}' is not in cluster-registry.yaml active clusters: {sorted(valid_clusters)}")

    if entry.get("lob", lob) not in VALID_LOBS:
        errors.append(f"lob must be one of {sorted(VALID_LOBS)}")

    if entry.get("environment") not in VALID_ENVIRONMENTS:
        errors.append(f"environment must be one of {sorted(VALID_ENVIRONMENTS)}")

    gov = entry.get("governance", {})
    if not re.match(r"^[0-9]{8}$", str(gov.get("map-id", ""))):
        errors.append("governance.map-id must be exactly 8 digits")
    if not re.match(r"^TO-[0-9]{6}$", str(gov.get("to-intake", ""))):
        errors.append("governance.to-intake must match TO-XXXXXX")
    if not re.match(r"^CHG-[A-Z0-9]+$", str(gov.get("change-ref", ""))):
        errors.append("governance.change-ref must match CHG-XXXXXXX")
    if not gov.get("team-name"):
        errors.append("governance.team-name is required")

    quota = entry.get("quota", DEFAULTS["quota"])
    if quota not in VALID_QUOTAS:
        errors.append(f"quota must be one of {sorted(VALID_QUOTAS)}")

    netpol = entry.get("netpol", DEFAULTS["netpol"])
    if netpol not in VALID_NETPOLS:
        errors.append(f"netpol must be one of {sorted(VALID_NETPOLS)}")

    data_class = gov.get("data-classification", DEFAULTS["data-classification"])
    if data_class not in VALID_DATA_CLASS:
        errors.append(f"data-classification must be one of {sorted(VALID_DATA_CLASS)}")

    for r in entry.get("roles", []):
        if not r.get("group"):
            errors.append("each role entry must have a non-empty 'group'")
        if r.get("role") not in ROLE_MAP and r.get("role") is not None:
            errors.append(f"role '{r.get('role')}' must be one of {list(ROLE_MAP.keys())}")

    ms = entry.get("machineSet")
    if isinstance(ms, dict):
        if ms.get("instanceType") not in APPROVED_SKUS:
            errors.append(f"machineSet.instanceType must be one of {sorted(APPROVED_SKUS)}")
        if not (1 <= int(ms.get("replicas", 0)) <= 10):
            errors.append("machineSet.replicas must be 1-10")
        if str(ms.get("zone", "")) not in {"1", "2", "3"}:
            errors.append("machineSet.zone must be '1', '2', or '3'")

    return errors


# ---------------------------------------------------------------------------
# Load existing RIR
# ---------------------------------------------------------------------------
def load_existing_rir(rir_path: str):
    if os.path.exists(rir_path):
        with open(rir_path) as f:
            return yaml.safe_load(f)
    return None


# ---------------------------------------------------------------------------
# Detect changed resource types
# ---------------------------------------------------------------------------
def detect_changed_resources(old_rir, new_rir: dict) -> list:
    if old_rir is None:
        changed = ["namespace", "quota", "limitrange", "netpol"]
        if new_rir.get("rbac", {}).get("roleBindings"):
            changed.append("rbac")
        if new_rir.get("machineSet"):
            changed.append("machineSet")
        return changed

    changed = []
    def differs(a, b):
        return json.dumps(a, sort_keys=True) != json.dumps(b, sort_keys=True)

    if differs(old_rir.get("namespace"), new_rir.get("namespace")):
        changed.append("namespace")
    if differs(old_rir.get("quota"), new_rir.get("quota")) or \
       differs(old_rir.get("limitRange"), new_rir.get("limitRange")):
        changed.append("quota")
        changed.append("limitrange")
    if differs(old_rir.get("networkPolicy"), new_rir.get("networkPolicy")):
        changed.append("netpol")
    if differs(old_rir.get("rbac"), new_rir.get("rbac")):
        changed.append("rbac")
    if differs(old_rir.get("machineSet"), new_rir.get("machineSet")):
        changed.append("machineSet")
    return changed


# ---------------------------------------------------------------------------
# Build RIR (#14: data-classification defaults to 'internal')
# ---------------------------------------------------------------------------
def build_rir(entry: dict, lob: str, existing_rir, author: str) -> dict:
    ns      = entry["namespace"]
    gov     = entry.get("governance", {})
    quota   = entry.get("quota",  DEFAULTS["quota"])
    netpol  = entry.get("netpol", DEFAULTS["netpol"])
    version = (existing_rir or {}).get("metadata", {}).get("version", 0) + 1

    # Default data-classification to 'internal' if not provided (#14)
    data_class = gov.get("data-classification", DEFAULTS["data-classification"])

    # Merge roles from existing RIR with patch
    existing_roles_input = []
    if existing_rir:
        for r in existing_rir.get("rbac", {}).get("roleBindings", []):
            role_key = next((k for k, v in ROLE_MAP.items() if v == r["clusterRole"]), None)
            if role_key:
                existing_roles_input.append({"group": r["group"], "role": role_key})
    merged_roles = merge_roles(existing_roles_input, entry.get("roles", []))

    # Merge custom labels
    existing_labels = (existing_rir or {}).get("namespace", {}).get("customLabels", {})
    merged_labels   = merge_labels(existing_labels, entry.get("labels", {}))

    rir = {
        "apiVersion": "platform.company-xxx.com/v1",
        "kind":       "ResourceIntentRecord",
        "metadata": {
            "name":      ns,
            "version":   version,
            "createdBy": author,
            "changeRef": gov.get("change-ref", ""),
        },
        "target": {
            "cluster":     entry.get("cluster", ""),
            "lob":         entry.get("lob", lob),
            "environment": entry.get("environment", ""),
        },
        "governance": {
            "mapId":              str(gov.get("map-id", "")),
            "toIntakeNumber":     gov.get("to-intake", ""),
            "teamName":           gov.get("team-name", ""),
            "dataClassification": data_class,
        },
        "lifecycle": {
            "state":         "active",
            "retentionDays": 365,
        },
        "namespace": {
            "name":        ns,
            "displayName": ns.replace("-", " ").title(),
        },
        "quota":         {"profile": quota},
        "limitRange":    {"profile": LIMITRANGE_FROM_QUOTA.get(quota, "standard")},
        "networkPolicy": {"template": netpol},
    }

    if gov.get("cost-center"):
        rir["governance"]["costCenter"] = gov["cost-center"]
    if merged_labels:
        rir["namespace"]["customLabels"] = merged_labels
    if merged_roles:
        rir["rbac"] = {
            "roleBindings": [
                {"group": r["group"], "clusterRole": ROLE_MAP[r["role"]]}
                for r in merged_roles if r.get("role") is not None
            ]
        }

            "subnet":        ms.get("subnet", "worker-subnet"),  # #9
            "resourceGroup": ms.get("resourceGroup", ""),        # #9
        }

    return rir

# Write audit log
# ---------------------------------------------------------------------------
def write_audit_log(audit_path: str, rir: dict, changed_resources: list,
                    author: str, input_file: str, is_new: bool) -> None:
    now = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    entries = []
    if os.path.exists(audit_path):
        with open(audit_path) as f:
            existing = yaml.safe_load(f) or {}
            entries = existing.get("changelog", [])
    entries.append({
        "version":          rir["metadata"]["version"],
        "timestamp":        now,
        "action":           "created" if is_new else "updated",
        "author":           author,
        "changeRef":        rir["metadata"].get("changeRef", ""),
        "changedResources": sorted(changed_resources),
        "sourceFile":       input_file,
    })
    header = (
        f"# Audit log for namespace: {rir['namespace']['name']}\n"
        f"# AUTO-GENERATED by input_to_rir.py - DO NOT EDIT MANUALLY\n"
        f"# Last updated: {now}\n"
    )
    with open(audit_path, "w") as f:
        f.write(header)
        yaml.dump({"namespace": rir["namespace"]["name"], "changelog": entries},
                  f, default_flow_style=False, allow_unicode=True, sort_keys=False)

def write_rir(rir_path: str, rir: dict, input_file: str) -> None:
    now = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")

# ---------------------------------------------------------------------------
def main() -> None:
    parser = argparse.ArgumentParser(description="Convert LOB input.yaml to per-namespace RIRs")
    parser.add_argument("--input",    required=True,  help="onboarding/<lob>.input.yaml")
    parser.add_argument("--schema",   required=True,  help="rir-v1-schema.json")
    parser.add_argument("--registry", required=True,  help="clusters/cluster-registry.yaml")
    parser.add_argument("--outdir",   required=True,  help="resource-catalog/")
    parser.add_argument("--author",   default="gitops-bot")
    parser.add_argument("--manifest", required=True,  help="output render manifest JSON")
    args = parser.parse_args()

    lob = os.path.basename(args.input).replace(".input.yaml", "")

    with open(args.input) as f:
        lob_input = yaml.safe_load(f)

    namespaces = lob_input.get("namespaces", [])
    if not namespaces:
        print(f"No namespaces found in {args.input}")
        sys.exit(0)

    # Load cluster registry once (#10)
    valid_clusters = load_cluster_registry(args.registry)

    manifest   = []
    has_errors = False

    for entry in namespaces:
        ns = entry.get("namespace", "<unknown>")
        print(f"Processing: {ns}")

        # Cross-LOB duplicate check (#6)
        duplicates = check_duplicate_namespace(ns, lob, args.outdir)
        if duplicates:
            print(f"  ERROR: namespace '{ns}' already exists in LOB(s): {', '.join(duplicates)}", file=sys.stderr)
            has_errors = True
            continue

        errors = validate_entry(entry, lob, valid_clusters)
        if errors:
            for e in errors:
                print(f"  ERROR: {e}", file=sys.stderr)
            has_errors = True
            continue

        rir_path   = os.path.join(args.outdir, lob, f"{ns}.rir.yaml")
        audit_path = os.path.join(args.outdir, lob, f"{ns}.audit.yaml")
        existing   = load_existing_rir(rir_path)
        is_new     = existing is None
        new_rir    = build_rir(entry, lob, existing, args.author)
        changed    = detect_changed_resources(existing, new_rir)

        if not changed and not is_new:
            print(f"  No changes - skipping")
            continue

        write_rir(rir_path, new_rir, args.input)
        write_audit_log(audit_path, new_rir, changed, args.author, args.input, is_new)
        print(f"  {'Created' if is_new else 'Updated'} v{new_rir['metadata']['version']} - changed: {changed}")

        manifest.append({
            "namespace":         ns,
            "lob":               lob,
            "rir_path":          rir_path,
            "audit_path":        audit_path,
            "changed_resources": changed,
            "is_new":            is_new,
        })

    if has_errors:
        print("Validation errors found - aborting.", file=sys.stderr)
        sys.exit(1)

    with open(args.manifest, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Manifest: {args.manifest} | Processed: {len(namespaces)} | Changed: {len(manifest)}")


if __name__ == "__main__":
    main()

"""

def build_context(rir: dict) -> dict:
    quota_profile = rir.get("quota", {}).get("profile", "small")
    lr_profile    = rir.get("limitRange", {}).get("profile", "standard")
    gov           = rir.get("governance", {})

        "display_name":        rir["namespace"].get("displayName", ns),

        "lr_profile":          lr_profile,
        **LIMITRANGE_PROFILES.get(lr_profile, LIMITRANGE_PROFILES["standard"]),

        "machineset_resource_group": ms.get("resourceGroup", "")          if ms else "",

        "rir_version":         rir.get("metadata", {}).get("version", 1),
        "change_ref":          rir.get("metadata", {}).get("changeRef", ""),
    }


def render_namespace(rir: dict, changed_resources: list, templates_dir: str, outdir: str) -> list:
    env = Environment(loader=FileSystemLoader(templates_dir),
                      undefined=StrictUndefined, trim_blocks=True, lstrip_blocks=True)
    ctx     = build_context(rir)
    cluster = rir["target"]["cluster"]
    lob     = rir["target"]["lob"]
    ns      = rir["namespace"]["name"]
    rendered = []

    for resource in changed_resources:
        if resource not in RESOURCE_TEMPLATE_MAP:
            print(f"  WARNING: no template for '{resource}' - skipping")
            continue
        tmpl_file, subdir = RESOURCE_TEMPLATE_MAP[resource]
        tmpl_path = os.path.join(templates_dir, tmpl_file)
        if not os.path.exists(tmpl_path):
            print(f"  WARNING: template not found: {tmpl_path} - skipping")
            continue
        out_path = os.path.join(outdir, cluster, "lob", lob, subdir, f"{ns}.yaml")
        os.makedirs(os.path.dirname(out_path), exist_ok=True)
        with open(out_path, "w") as f:
            f.write(env.get_template(tmpl_file).render(**ctx))
        print(f"  Rendered [{resource}]: {out_path}")
        rendered.append(out_path)

    return rendered


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest",  required=True)
    parser.add_argument("--templates", required=True)
    parser.add_argument("--outdir",    required=True)
    args = parser.parse_args()

    with open(args.manifest) as f:
        manifest = json.load(f)

    if not manifest:
        print("No changes in manifest - nothing to render.")
        return

    total = []
    for item in manifest:
        ns       = item["namespace"]
        rir_path = item["rir_path"]
        changed  = item["changed_resources"]
        print(f"Rendering: {ns} ({len(changed)} resource types)")
        if not os.path.exists(rir_path):
            print(f"  ERROR: RIR not found: {rir_path}", file=sys.stderr)
            continue
        with open(rir_path) as f:
            rir = yaml.safe_load(f)
        total.extend(render_namespace(rir, changed, args.templates, args.outdir))

    print(f"Render complete. Files written: {len(total)}")


if __name__ == "__main__":
    main()

## Create GITOPS_BOT_TOKEN

### Option 1: Fine-grained Personal Access Token (Minimum viable)
1. GitHub Settings > Developer Settings > Personal Access Tokens > Fine-grained
2. Repository: ocp-gitops-platform only
3. Permissions:
   - Contents: Read and Write
   - Pull Requests: Read and Write
4. Add as repository secret: Settings > Secrets > Actions > New secret

### Option 2: GitHub App (Recommended for production)
1. GitHub Settings > Developer Settings > GitHub Apps > New GitHub App
2. Permissions:
   - Repository contents: Read and Write
   - Pull Requests: Read and Write
3. Install on organisation, restrict to ocp-gitops-platform repo
4. Generate private key, store in vault
5. Use a GitHub App token action to generate short-lived tokens per run
