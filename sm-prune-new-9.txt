
# Check what ConfigMap patterns exist in openshift-apiserver
oc get cm -n openshift-apiserver -o json | \
jq -r '.items[] | .metadata.name' | \
grep -E "^(revision-status|audit|config)-[0-9]+$" | \
sed 's/-[0-9]*$//' | sort | uniq -c | sort -rn



# Check actual secret types
oc get secrets -n openshift-kube-apiserver -o json | \
  jq -r '.items[] | select(.metadata.name | test("localhost-recovery-client")) | 
         "\(.metadata.name) | type=\(.type) | created=\(.metadata.creationTimestamp)"' | head -10

oc get secrets -n openshift-etcd -o json | \
  jq -r '.items[] | select(.metadata.name | test("etcd-client")) | 
         "\(.metadata.name) | type=\(.type) | created=\(.metadata.creationTimestamp)"' | head -10

# Count how many exist (to understand rotation volume)
oc get secrets -n openshift-kube-apiserver --no-headers | grep "localhost-recovery-client" | wc -l
oc get secrets -n openshift-etcd --no-headers | grep "etcd-client" | wc -l
