
bash# Find last Completed revision in openshift-oauth-apiserver
oc get cm -n openshift-oauth-apiserver -o json | \
  jq '[.items[] | select(.metadata.name | test("^revision-status-[0-9]+$")) |
    select((.data["reason"] // "") | test("changed|InProgress|Abandoned|Failed") | not) |
    {name: .metadata.name, rev: (.data["revision"] | tonumber), reason: .data["reason"], ts: .metadata.creationTimestamp}] |
    sort_by(.rev) | last'
